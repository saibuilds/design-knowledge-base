Spotlight

Mouse-following spotlight effect — the #1 most-used Aceternity component for hero sections.

jsx
import { Spotlight } from '@/components/ui/spotlight';

export function HeroSection() {
  return (
    <div className="relative min-h-screen overflow-hidden bg-black/[0.96] antialiased">
      <Spotlight
        className="-top-40 left-0 md:-top-20 md:left-60"
        fill="white"  // spotlight color
      />
      <div className="relative z-10 flex flex-col items-center justify-center min-h-screen">
        <h1 className="text-6xl font-bold text-white">
          Build something amazing
        </h1>
      </div>
    </div>
  );
}

Card Hover Effect

Animated card grid with hover reveal — great for feature lists and portfolio grids.

jsx
import { HoverEffect } from '@/components/ui/card-hover-effect';

const items = [
  { title: 'Speed', description: 'Build 10x faster with our components.', link: '#' },
  { title: 'Scale', description: 'From prototype to production effortlessly.', link: '#' },
  { title: 'Security', description: 'Enterprise-grade security built in.', link: '#' },
];

export function Features() {
  return <HoverEffect items={items} className="max-w-4xl mx-auto" />;
}

Background Effects

jsx — wavy, aurora, beams, vortex
import { WavyBackground } from '@/components/ui/wavy-background';
import { AuroraBackground } from '@/components/ui/aurora-background';
import { BackgroundBeams } from '@/components/ui/background-beams';
import { Vortex } from '@/components/ui/vortex';
import { BackgroundGradient } from '@/components/ui/background-gradient';

// Wavy canvas background
<WavyBackground className="max-w-4xl mx-auto pb-40" colors={['#38bdf8', '#818cf8', '#c084fc']}>
  <h1 className="text-4xl font-bold text-white text-center">Hero Text</h1>
</WavyBackground>

// Aurora borealis
<AuroraBackground>
  <div className="relative z-10">Content</div>
</AuroraBackground>

// Beam paths background
<div className="relative min-h-screen">
  <BackgroundBeams />
  <div className="relative z-10">Content</div>
</div>

// Vortex swirl
<Vortex backgroundColor="black" className="min-h-screen">
  <h1 className="text-white text-5xl font-bold">Into the vortex</h1>
</Vortex>

// Glowing border card
<BackgroundGradient className="rounded-[22px] p-4 bg-zinc-900">
  <p className="text-white">Glowing card content</p>
</BackgroundGradient>

Text Effects

jsx
import { TextGenerateEffect } from '@/components/ui/text-generate-effect';
import { TypewriterEffect } from '@/components/ui/typewriter-effect';
import { FlipWords } from '@/components/ui/flip-words';

// Word-by-word fade in
const words = 'Build beautiful websites with ease.';
<TextGenerateEffect words={words} className="text-4xl font-bold" />

// Typewriter with colored highlight
const words2 = [
  { text: 'Build' },
  { text: 'awesome' },
  { text: 'apps' },
  { text: 'with', className: 'text-blue-500' },
  { text: 'Aceternity.', className: 'text-blue-500' },
];
<TypewriterEffect words={words2} className="text-4xl font-bold" />

// Rotating/flipping words
const flip = ['beautiful', 'modern', 'fast', 'reliable'];
<h1>Build <FlipWords words={flip} /> websites</h1>

Moving Border Button

CTA button with animated gradient border — one of the most-copied Aceternity components.

jsx
import { MovingBorder } from '@/components/ui/moving-border';

<MovingBorder duration={3000} rx="30%" ry="30%">
  <button className="bg-black text-white px-8 py-3 font-semibold rounded-full">
    Get Started
  </button>
</MovingBorder>

// Or use the pre-built Button variant
import { Button } from '@/components/ui/moving-border';
<Button
  borderRadius="1.75rem"
  className="bg-black text-white border-slate-800"
>
  Border button
</Button>

Placeholder & Vanish Input

Input field that animates through placeholder suggestions and vanishes on submit.

jsx
import { PlaceholderAndVanishInput } from '@/components/ui/placeholder-and-vanish-input';

const placeholders = [
  'What is the meaning of life?',
  'Who painted the Mona Lisa?',
  'What is the speed of light?',
];

<PlaceholderAndVanishInput
  placeholders={placeholders}
  onChange={e => console.log(e.target.value)}
  onSubmit={e => { e.preventDefault(); console.log('submitted'); }}
/>
