Usage — Copy-Paste Pattern

// Animata components are copy-paste — visit animata.design, pick a component,
// copy the source into your project at components/animata/[name].tsx

// Example: Typing Animation component
"use client";
import { useEffect, useState } from "react";
import { cn } from "@/lib/utils";

interface TypingAnimationProps {
  text: string;
  className?: string;
  duration?: number;
}

export function TypingAnimation({ text, className, duration = 100 }: TypingAnimationProps) {
  const [displayed, setDisplayed] = useState("");

  useEffect(() => {
    let i = 0;
    const timer = setInterval(() => {
      if (i < text.length) {
        setDisplayed(text.slice(0, i + 1));
        i++;
      } else {
        clearInterval(timer);
      }
    }, duration);
    return () => clearInterval(timer);
  }, [text, duration]);

  return (
    <span className={cn("font-mono", className)}>
      {displayed}
      <span className="animate-pulse">|</span>
    </span>
  );
}

// Usage
<TypingAnimation text="Hello, World!" className="text-4xl font-bold" />

Stagger Children Animation

import { motion } from "framer-motion";

const container = {
  hidden: { opacity: 0 },
  show: {
    opacity: 1,
    transition: { staggerChildren: 0.1 }
  }
};

const item = {
  hidden: { opacity: 0, y: 20 },
  show: { opacity: 1, y: 0, transition: { type: "spring", stiffness: 300, damping: 24 } }
};

export function StaggerList({ items }: { items: string[] }) {
  return (
    <motion.ul variants={container} initial="hidden" animate="show">
      {items.map((text, i) => (
        <motion.li key={i} variants={item} className="p-3 border-b border-border">
          {text}
        </motion.li>
      ))}
    </motion.ul>
  );
}

Text Effect Component

"use client";
import { motion, Variants } from "framer-motion";

type TextEffectProps = {
  children: string;
  per?: "word" | "char";
  preset?: "fade" | "slide" | "scale";
  delay?: number;
  className?: string;
};

const presets: Record = {
  fade: {
    hidden: { opacity: 0 },
    visible: { opacity: 1 },
  },
  slide: {
    hidden: { opacity: 0, y: 20 },
    visible: { opacity: 1, y: 0 },
  },
  scale: {
    hidden: { opacity: 0, scale: 0.5 },
    visible: { opacity: 1, scale: 1 },
  },
};

export function TextEffect({
  children,
  per = "word",
  preset = "slide",
  delay = 0,
  className,
}: TextEffectProps) {
  const segments = per === "char" ? children.split("") : children.split(" ");
  const variants = presets[preset];

  return (
    <motion.p
      className={className}
      initial="hidden"
      animate="visible"
      transition={{ staggerChildren: 0.05, delayChildren: delay }}
    >
      {segments.map((seg, i) => (
        <motion.span key={i} variants={variants} style={{ display: "inline-block" }}>
          {seg}{per === "word" ? " " : ""}
        </motion.span>
      ))}
    </motion.p>
  );
}

// Usage
<TextEffect preset="slide" per="word" className="text-3xl font-bold">
  Animate each word on mount
</TextEffect>

Infinite Scroll Ticker

// Skewed infinite scroll — typical SyntaxUI pattern
"use client";
import { cn } from "@/lib/utils";

export function InfiniteScroll({
  items,
  speed = 30,
  className,
}: {
  items: string[];
  speed?: number;
  className?: string;
}) {
  return (
    <div
      className={cn("flex overflow-hidden [mask-image:linear-gradient(to_right,transparent,white_20%,white_80%,transparent)]", className)}
    >
      {[0, 1].map((i) => (
        <div
          key={i}
          className="flex shrink-0 gap-4 py-4 animate-marquee"
          style={{ "--duration": `${speed}s` } as React.CSSProperties}
        >
          {items.map((item, j) => (
            <span key={j} className="px-4 py-1 bg-muted rounded-full text-sm font-medium">
              {item}
            </span>
          ))}
        </div>
      ))}
    </div>
  );
}

Shift Card (Hover Reveal)

"use client";
import { motion } from "framer-motion";
import { useState } from "react";

interface ShiftCardProps {
  title: string;
  description: string;
  detail: string;
  icon?: React.ReactNode;
}

export function ShiftCard({ title, description, detail, icon }: ShiftCardProps) {
  const [hovered, setHovered] = useState(false);

  return (
    <motion.div
      className="relative w-64 h-40 bg-card border border-border rounded-xl p-5 cursor-pointer overflow-hidden"
      onHoverStart={() => setHovered(true)}
      onHoverEnd={() => setHovered(false)}
    >
      {/* Default state */}
      <motion.div
        animate={{ opacity: hovered ? 0 : 1, y: hovered ? -10 : 0 }}
        transition={{ duration: 0.2 }}
      >
        {icon && <div className="mb-2 text-muted-foreground">{icon}</div>}
        <h3 className="font-semibold text-foreground">{title}</h3>
        <p className="text-sm text-muted-foreground mt-1">{description}</p>
      </motion.div>

      {/* Hover detail */}
      <motion.div
        className="absolute inset-0 p-5 flex items-center justify-center bg-card"
        animate={{ opacity: hovered ? 1 : 0, y: hovered ? 0 : 10 }}
        transition={{ duration: 0.2 }}
      >
        <p className="text-sm text-foreground text-center">{detail}</p>
      </motion.div>
    </motion.div>
  );
}

// Password Input with visibility toggle
import { useState } from "react";
import { Eye, EyeOff } from "lucide-react";
import { Button } from "@/components/ui/button";

export function PasswordInput() {
  const [show, setShow] = useState(false);
  return (
    <div className="relative">
      <Input type={show ? "text" : "password"} placeholder="Password" className="pr-10" />
      <Button
        variant="ghost"
        size="icon"
        className="absolute right-0 top-0 h-full w-10"
        onClick={() => setShow(!show)}
      >
        {show ? <EyeOff size={16} /> : <Eye size={16} />}
      </Button>
    </div>
  );
}
