EncryptButton

Free Text Animation

On hover, letters scramble through random characters before resolving back to the target text. Creates a hacker/terminal aesthetic.

// EncryptButton.tsx
"use client"
import { useRef, useState } from "react"

const CYCLES_PER_LETTER = 2
const SHUFFLE_TIME = 50
const CHARS = "!@#$%^&*():{};|,.<>/?"

interface EncryptButtonProps {
  targetText: string
}

export const EncryptButton = ({ targetText }: EncryptButtonProps) => {
  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null)
  const [text, setText] = useState(targetText)

  const scramble = () => {
    let pos = 0

    intervalRef.current = setInterval(() => {
      const scrambled = targetText
        .split("")
        .map((char, index) => {
          if (pos / CYCLES_PER_LETTER > index) {
            return char
          }
          const randomChar = CHARS[Math.floor(Math.random() * CHARS.length)]
          return randomChar
        })
        .join("")

      setText(scrambled)
      pos++

      if (pos >= targetText.length * CYCLES_PER_LETTER) {
        stopScramble()
      }
    }, SHUFFLE_TIME)
  }

  const stopScramble = () => {
    clearInterval(intervalRef.current ?? undefined)
    setText(targetText)
  }

  return (
    <button
      onMouseEnter={scramble}
      onMouseLeave={stopScramble}
      className="group relative overflow-hidden rounded-lg border border-neutral-500 bg-neutral-700 px-4 py-2 font-mono font-medium uppercase text-neutral-300 transition-colors hover:text-indigo-300"
    >
      <div className="relative z-10 flex items-center gap-2">
        <span>{text}</span>
      </div>
      <motion.div
        initial={{ y: "100%" }}
        animate={{ y: "-100%" }}
        transition={{ repeat: Infinity, repeatType: "loop", duration: 1, ease: "linear" }}
        className="absolute inset-0 z-0 scale-125 bg-gradient-to-t from-indigo-400/0 from-40% via-indigo-400/100 to-indigo-400/0 to-60% opacity-0 transition-opacity duration-300 group-hover:opacity-100"
      />
    </button>
  )
}

BubbleButton

Free Bubbles Animation

Clicking the button spawns small circular bubble particles that float upward and fade out.

// BubbleButton.tsx
"use client"
import { useRef, useState } from "react"
import { motion, AnimatePresence } from "framer-motion"

interface Bubble {
  id: number
  x: number
  y: number
  size: number
}

interface BubbleButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  children: React.ReactNode
}

export function BubbleButton({ children, onClick, className, ...props }: BubbleButtonProps) {
  const [bubbles, setBubbles] = useState<Bubble[]>([])
  const containerRef = useRef<HTMLButtonElement>(null)

  const handleClick = (e: React.MouseEvent<HTMLButtonElement>) => {
    const rect = containerRef.current!.getBoundingClientRect()
    const x = e.clientX - rect.left
    const y = e.clientY - rect.top

    const newBubbles: Bubble[] = Array.from({ length: 6 }, (_, i) => ({
      id: Date.now() + i,
      x: x + (Math.random() - 0.5) * 60,
      y: y + (Math.random() - 0.5) * 20,
      size: Math.random() * 12 + 6,
    }))

    setBubbles((prev) => [...prev, ...newBubbles])
    setTimeout(() => {
      setBubbles((prev) => prev.filter((b) => !newBubbles.find((nb) => nb.id === b.id)))
    }, 700)

    onClick?.(e)
  }

  return (
    <button
      ref={containerRef}
      onClick={handleClick}
      className={`relative overflow-hidden rounded-full bg-indigo-600 px-6 py-2.5 text-sm font-semibold text-white hover:bg-indigo-500 transition-colors active:scale-95 ${className ?? ""}`}
      {...props}
    >
      {children}
      <AnimatePresence>
        {bubbles.map((bubble) => (
          <motion.span
            key={bubble.id}
            initial={{ opacity: 0.8, scale: 0, x: bubble.x, y: bubble.y }}
            animate={{ opacity: 0, scale: 1, y: bubble.y - 60 }}
            exit={{ opacity: 0 }}
            transition={{ duration: 0.6, ease: "easeOut" }}
            style={{
              position: "absolute",
              width: bubble.size,
              height: bubble.size,
              borderRadius: "50%",
              background: "rgba(255,255,255,0.6)",
              pointerEvents: "none",
              translateX: "-50%",
              translateY: "-50%",
            }}
          />
        ))}
      </AnimatePresence>
    </button>
  )
}

SplashButton

Free Ripple Animation

Click creates a radial ripple/splash wave that expands from the click point and fades out.

// SplashButton.tsx
"use client"
import { useState } from "react"
import { motion, AnimatePresence } from "framer-motion"

interface Ripple {
  id: number
  x: number
  y: number
}

interface SplashButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  children: React.ReactNode
  color?: string
}

export function SplashButton({ children, onClick, color = "rgba(255,255,255,0.35)", className, ...props }: SplashButtonProps) {
  const [ripples, setRipples] = useState<Ripple[]>([])

  const handleClick = (e: React.MouseEvent<HTMLButtonElement>) => {
    const rect = e.currentTarget.getBoundingClientRect()
    const newRipple: Ripple = {
      id: Date.now(),
      x: e.clientX - rect.left,
      y: e.clientY - rect.top,
    }
    setRipples((r) => [...r, newRipple])
    setTimeout(() => setRipples((r) => r.filter((rp) => rp.id !== newRipple.id)), 800)
    onClick?.(e)
  }

  return (
    <button
      onClick={handleClick}
      className={`relative overflow-hidden rounded-xl bg-violet-600 px-6 py-3 font-semibold text-white hover:bg-violet-500 transition-colors ${className ?? ""}`}
      {...props}
    >
      {children}
      <AnimatePresence>
        {ripples.map((ripple) => (
          <motion.span
            key={ripple.id}
            initial={{ opacity: 1, scale: 0, x: ripple.x, y: ripple.y, width: 0, height: 0 }}
            animate={{ opacity: 0, scale: 1, width: 300, height: 300 }}
            exit={{ opacity: 0 }}
            transition={{ duration: 0.7, ease: "easeOut" }}
            style={{
              position: "absolute",
              background: color,
              borderRadius: "50%",
              transform: "translate(-50%, -50%)",
              pointerEvents: "none",
            }}
          />
        ))}
      </AnimatePresence>
    </button>
  )
}

GlitchButton

Free Glitch CSS

On hover, button text creates RGB-split glitch layers that shift horizontally, simulating a VHS/glitch effect.

"use client"
import { useState } from "react"
import { motion } from "framer-motion"

interface GlitchButtonProps {
  children: string
  className?: string
}

export function GlitchButton({ children, className }: GlitchButtonProps) {
  const [hovered, setHovered] = useState(false)

  return (
    <button
      onMouseEnter={() => setHovered(true)}
      onMouseLeave={() => setHovered(false)}
      className={`relative overflow-hidden rounded-lg border border-white/20 bg-neutral-900 px-6 py-3 font-mono font-bold uppercase text-white ${className ?? ""}`}
    >
      {/* Main text */}
      <span className="relative z-10">{children}</span>

      {/* Red ghost layer */}
      <motion.span
        aria-hidden
        animate={hovered ? { x: [0, -4, 3, -2, 0], opacity: [0, 0.8, 0.8, 0.8, 0] } : { x: 0, opacity: 0 }}
        transition={{ duration: 0.4, times: [0, 0.2, 0.5, 0.8, 1] }}
        className="absolute inset-0 flex items-center justify-center font-mono font-bold uppercase"
        style={{ color: "#ff0000", mixBlendMode: "screen", clipPath: "inset(30% 0 30% 0)" }}
      >
        {children}
      </motion.span>

      {/* Blue ghost layer */}
      <motion.span
        aria-hidden
        animate={hovered ? { x: [0, 4, -3, 2, 0], opacity: [0, 0.8, 0.8, 0.8, 0] } : { x: 0, opacity: 0 }}
        transition={{ duration: 0.4, times: [0, 0.2, 0.5, 0.8, 1], delay: 0.05 }}
        className="absolute inset-0 flex items-center justify-center font-mono font-bold uppercase"
        style={{ color: "#00ffff", mixBlendMode: "screen", clipPath: "inset(60% 0 5% 0)" }}
      >
        {children}
      </motion.span>
    </button>
  )
}

MagnetButton

Free Magnetic Physics

The button (and optionally its inner label) moves toward the mouse cursor as it approaches, snapping back with a spring when the cursor leaves. Classic magnetic cursor effect.

// MagnetButton.tsx
"use client"
import { useRef, useState } from "react"
import { motion, useSpring, useTransform, MotionValue } from "framer-motion"

function useMagnet(strength = 0.4) {
  const ref = useRef<HTMLButtonElement>(null)
  const x = useSpring(0, { stiffness: 200, damping: 20 })
  const y = useSpring(0, { stiffness: 200, damping: 20 })

  const onMouseMove = (e: React.MouseEvent) => {
    const el = ref.current
    if (!el) return
    const rect = el.getBoundingClientRect()
    const cx = rect.left + rect.width / 2
    const cy = rect.top + rect.height / 2
    x.set((e.clientX - cx) * strength)
    y.set((e.clientY - cy) * strength)
  }

  const onMouseLeave = () => {
    x.set(0)
    y.set(0)
  }

  return { ref, x, y, onMouseMove, onMouseLeave }
}

interface MagnetButtonProps {
  children: React.ReactNode
  className?: string
  strength?: number
}

export function MagnetButton({ children, className, strength = 0.4 }: MagnetButtonProps) {
  const { ref, x, y, onMouseMove, onMouseLeave } = useMagnet(strength)

  return (
    <motion.button
      ref={ref}
      style={{ x, y }}
      onMouseMove={onMouseMove}
      onMouseLeave={onMouseLeave}
      className={`relative rounded-full bg-white px-8 py-4 font-semibold text-black transition-shadow hover:shadow-xl ${className ?? ""}`}
    >
      <motion.span
        style={{ x: useTransform(x, (v) => v * 0.4), y: useTransform(y, (v) => v * 0.4) }}
        className="block"
      >
        {children}
      </motion.span>
    </motion.button>
  )
}

FlipButton

Free 3D Hover

Button with two faces that flips on hover, revealing alternate text or content on the back face using 3D CSS transforms.

// FlipButton.tsx
"use client"
import { motion } from "framer-motion"

interface FlipButtonProps {
  frontText: string
  backText: string
  frontClass?: string
  backClass?: string
}

export function FlipButton({ frontText, backText, frontClass, backClass }: FlipButtonProps) {
  return (
    <motion.button
      initial="rest"
      whileHover="hover"
      className="relative overflow-hidden rounded-full px-6 py-2.5 font-semibold"
      style={{ perspective: 600 }}

      {/* Front face */}
      <motion.span
        variants={{

      >
        {frontText}
      </motion.span>
      {/* Back face */}
      <motion.span
        variants={{
          rest: { rotateX: 90, y: -10, opacity: 0 },
          hover: { rotateX: 0, y: "-100%", opacity: 1 },
        }}
        transition={{ duration: 0.3, ease: "easeInOut" }}
        className={`absolute inset-0 flex items-center justify-center ${backClass ?? "bg-indigo-600 text-white"}`}
        style={{ backfaceVisibility: "hidden", borderRadius: 9999 }}
      >
        {backText}
      </motion.span>
    </motion.button>
  )
}

CountdownButton

Free Interaction

Button that requires the user to hold it down for a countdown before confirming — prevents accidental destructive actions.

// CountdownButton.tsx
"use client"
import { useEffect, useRef, useState } from "react"
import { motion } from "framer-motion"

interface CountdownButtonProps {
  holdDuration?: number
  onConfirm: () => void
  children: React.ReactNode
}

export function CountdownButton({ holdDuration = 2000, onConfirm, children }: CountdownButtonProps) {
  const [holding, setHolding] = useState(false)
  const [progress, setProgress] = useState(0)
  const startTime = useRef<number | null>(null)
  const rafRef = useRef<number>(0)

  const startHold = () => {
    setHolding(true)
    startTime.current = performance.now()
    const tick = () => {
      const elapsed = performance.now() - (startTime.current ?? 0)
      const pct = Math.min(elapsed / holdDuration, 1)
      setProgress(pct)
      if (pct < 1) {
        rafRef.current = requestAnimationFrame(tick)
      } else {
        onConfirm()
        setHolding(false)
        setProgress(0)
      }
    }
    rafRef.current = requestAnimationFrame(tick)
  }

  const stopHold = () => {
    cancelAnimationFrame(rafRef.current)
    setHolding(false)
    setProgress(0)
  }

  return (
    <button
      onMouseDown={startHold}
      onMouseUp={stopHold}
      onMouseLeave={stopHold}
      onTouchStart={startHold}
      onTouchEnd={stopHold}
      className="relative overflow-hidden rounded-xl bg-red-600 px-6 py-3 font-semibold text-white select-none"
    >
      {/* Progress fill */}
      <motion.span
        className="pointer-events-none absolute inset-0 origin-left bg-red-400"
        style={{ scaleX: progress }}
      />
      <span className="relative z-10">
        {holding ? `Hold... ${Math.round(progress * 100)}%` : children}
      </span>
    </button>
  )
}

// <CountdownButton onConfirm={() => deleteItem()}>Hold to Delete</CountdownButton>

SwipeCards

Free Gesture Tinder-style

A draggable stack of cards. Drag a card past a threshold to dismiss it left or right. The card beneath becomes active. Tinder-style swipe UI.

// SwipeCards.tsx
"use client"
import { useState } from "react"
import { motion, useMotionValue, useTransform, AnimatePresence } from "framer-motion"

interface CardData {
  id: number
  content: string
  bg: string
}

const INITIAL_CARDS: CardData[] = [
  { id: 1, content: "Card One", bg: "#6366f1" },
  { id: 2, content: "Card Two", bg: "#8b5cf6" },
  { id: 3, content: "Card Three", bg: "#ec4899" },
  { id: 4, content: "Card Four", bg: "#f59e0b" },
]

function SwipeCard({
  card,
  onSwipe,
  isTop,
}: {
  card: CardData
  onSwipe: (id: number, dir: "left" | "right") => void
  isTop: boolean
}) {
  const x = useMotionValue(0)
  const rotate = useTransform(x, [-200, 200], [-25, 25])
  const opacity = useTransform(x, [-200, -100, 0, 100, 200], [0, 1, 1, 1, 0])

  const handleDragEnd = (_: unknown, info: { offset: { x: number } }) => {
    if (info.offset.x > 100) onSwipe(card.id, "right")
    else if (info.offset.x < -100) onSwipe(card.id, "left")
  }

  return (
    <motion.div
      drag={isTop ? "x" : false}
      dragConstraints={{ left: 0, right: 0 }}
      onDragEnd={handleDragEnd}
      style={{ x, rotate, opacity, background: card.bg }}
      className="absolute inset-0 flex items-center justify-center rounded-2xl text-2xl font-bold text-white shadow-2xl cursor-grab active:cursor-grabbing"
      whileDrag={{ scale: 1.05 }}
    >
      {card.content}
    </motion.div>
  )
}

export function SwipeCards() {
  const [cards, setCards] = useState(INITIAL_CARDS)

  const handleSwipe = (id: number) => {
    setCards((prev) => prev.filter((c) => c.id !== id))
  }

  return (
    <div className="relative h-64 w-48 mx-auto">
      <AnimatePresence>
        {cards.map((card, i) => (
          <SwipeCard
            key={card.id}
            card={card}
            isTop={i === cards.length - 1}
            onSwipe={handleSwipe}
          />
        ))}
      </AnimatePresence>
      {cards.length === 0 && (
        <button
          onClick={() => setCards(INITIAL_CARDS)}
          className="absolute inset-0 flex items-center justify-center rounded-2xl border-2 border-dashed border-neutral-600 text-neutral-400 text-sm"
        >
          Reset cards
        </button>
      )}
    </div>
  )
}

TiltCard

Free 3D Mouse Tracking

Card that tilts in 3D perspective based on mouse position. Inner elements can be given a translateZ offset to create a parallax depth effect.

// TiltCard.tsx
"use client"
import { useRef } from "react"
import { motion, useMotionValue, useSpring, useTransform } from "framer-motion"

interface TiltCardProps {
  children: React.ReactNode
  className?: string
  tiltStrength?: number
}

export function TiltCard({ children, className, tiltStrength = 15 }: TiltCardProps) {
  const cardRef = useRef<HTMLDivElement>(null)

  const rawX = useMotionValue(0)
  const rawY = useMotionValue(0)

  const x = useSpring(rawX, { stiffness: 300, damping: 30 })
  const y = useSpring(rawY, { stiffness: 300, damping: 30 })

  const rotateX = useTransform(y, [-0.5, 0.5], [tiltStrength, -tiltStrength])
  const rotateY = useTransform(x, [-0.5, 0.5], [-tiltStrength, tiltStrength])

  // Shine effect
  const shineX = useTransform(x, [-0.5, 0.5], ["140%", "-40%"])
  const shineY = useTransform(y, [-0.5, 0.5], ["140%", "-40%"])

  const handleMouseMove = (e: React.MouseEvent<HTMLDivElement>) => {
    const el = cardRef.current
    if (!el) return
    const rect = el.getBoundingClientRect()
    rawX.set((e.clientX - rect.left) / rect.width - 0.5)
    rawY.set((e.clientY - rect.top) / rect.height - 0.5)
  }

  const handleMouseLeave = () => {
    rawX.set(0)
    rawY.set(0)
  }

  return (
    <motion.div
      ref={cardRef}
      onMouseMove={handleMouseMove}
      onMouseLeave={handleMouseLeave}
      style={{ rotateX, rotateY, transformStyle: "preserve-3d" }}
      className={`relative overflow-hidden rounded-2xl border border-white/10 bg-neutral-900 p-6 shadow-2xl ${className ?? ""}`}
    >
      {/* Shine overlay */}
      <motion.div
        className="pointer-events-none absolute inset-0 rounded-2xl opacity-0 group-hover:opacity-100"
        style={{
          background: useTransform(
            [shineX, shineY],
            ([sx, sy]: string[]) => `radial-gradient(circle at ${sx} ${sy}, rgba(255,255,255,0.12), transparent 60%)`
          ),
        }}
      />
      <div style={{ transform: "translateZ(40px)" }}>
        {children}
      </div>
    </motion.div>
  )
}

HoverImageLinks

Free Image Reveal Navigation

A list of links where hovering over each reveals a floating image preview that follows the cursor. Made famous by awwwards-style agency sites.

// HoverImageLinks.tsx
"use client"
import { useRef, useState } from "react"
import { motion, AnimatePresence } from "framer-motion"

interface LinkItem {
  title: string
  href: string
  imgSrc: string
}

const links: LinkItem[] = [
  { title: "About", href: "/about", imgSrc: "https://images.unsplash.com/photo-1522071820081-009f0129c71c?w=400" },
  { title: "Work", href: "/work", imgSrc: "https://images.unsplash.com/photo-1558655146-364adaf1fcc9?w=400" },
  { title: "Contact", href: "/contact", imgSrc: "https://images.unsplash.com/photo-1521737711867-e3b97375f902?w=400" },
]

function LinkItem({ item }: { item: LinkItem }) {
  const [hovered, setHovered] = useState(false)
  const [pos, setPos] = useState({ x: 0, y: 0 })
  const ref = useRef<HTMLAnchorElement>(null)

  const handleMouseMove = (e: React.MouseEvent) => {
    setPos({ x: e.clientX, y: e.clientY })
  }

  return (
    <>
      <a
        ref={ref}
        href={item.href}
        onMouseEnter={() => setHovered(true)}
        onMouseLeave={() => setHovered(false)}
        onMouseMove={handleMouseMove}
        className="block border-b border-neutral-800 py-6 text-3xl font-bold text-neutral-300 transition-colors hover:text-white cursor-none"
      >
        {item.title}
      </a>
      <AnimatePresence>
        {hovered && (
          <motion.div
            initial={{ opacity: 0, scale: 0.8, rotate: -5 }}
            animate={{ opacity: 1, scale: 1, rotate: 0 }}
            exit={{ opacity: 0, scale: 0.8, rotate: 5 }}
            transition={{ duration: 0.2 }}
            style={{ left: pos.x + 20, top: pos.y - 80 }}
            className="pointer-events-none fixed z-50 h-40 w-56 overflow-hidden rounded-xl shadow-2xl"
          >
            <img src={item.imgSrc} alt={item.title} className="h-full w-full object-cover" />
          </motion.div>
        )}
      </AnimatePresence>
    </>
  )
}

export function HoverImageLinks() {
  return (
    <nav className="max-w-2xl mx-auto py-16 px-4">
      {links.map((item) => (
        <LinkItem key={item.title} item={item} />
      ))}
    </nav>
  )
}

ParallaxCard

Free Parallax Scroll

Card whose background image moves at a slower rate than the scroll speed, creating a parallax depth illusion.

// ParallaxCard.tsx
"use client"
import { useRef } from "react"
import { motion, useScroll, useTransform } from "framer-motion"

interface ParallaxCardProps {
  imgSrc: string
  title: string
  description?: string
}

export function ParallaxCard({ imgSrc, title, description }: ParallaxCardProps) {
  const ref = useRef<HTMLDivElement>(null)
  const { scrollYProgress } = useScroll({ target: ref, offset: ["start end", "end start"] })
  const imgY = useTransform(scrollYProgress, [0, 1], ["-20%", "20%"])

  return (
    <div ref={ref} className="relative h-80 overflow-hidden rounded-2xl">
      <motion.img
        src={imgSrc}
        alt={title}
        style={{ y: imgY, scale: 1.3 }}
        className="absolute inset-0 h-full w-full object-cover"
      />
      <div className="absolute inset-0 bg-gradient-to-t from-black/70 to-transparent" />
      <div className="absolute bottom-0 left-0 p-6">
        <h3 className="text-xl font-bold text-white">{title}</h3>
        {description && <p className="text-sm text-white/70 mt-1">{description}</p>}
      </div>
    </div>
  )
}

UnderlineToBackground

Free Link Effect

On hover, a colored underline expands upward to fill the full height of the text element, inverting text color — creating a highlight/background reveal effect.

// UnderlineToBackground.tsx
"use client"
import { motion } from "framer-motion"

interface UnderlineToBackgroundProps {
  children: string
  href?: string
  highlightColor?: string
  textColor?: string
  className?: string
}

export function UnderlineToBackground({
  children,
  href = "#",
  highlightColor = "#ffffff",
  textColor = "#000000",
  className,
}: UnderlineToBackgroundProps) {
  return (
    <motion.a
      href={href}
      initial="rest"
      whileHover="hover"
      className={`relative inline-block cursor-pointer overflow-hidden font-semibold ${className ?? ""}`}
    >
      {/* Highlight background that grows from bottom */}
      <motion.span
        variants={{
          rest: { scaleY: 0, originY: 1 },
          hover: { scaleY: 1, originY: 1 },
        }}

ScrambleHover

Free Text Effect

Text scrambles through random characters on hover, then resolves character by character from left to right. A more refined version of the EncryptButton effect applied to any text.

// ScrambleHover.tsx
"use client"
import { useRef, useState } from "react"

const CHARS = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789"
const SCRAMBLE_SPEED = 40
const RESOLVE_SPEED = 60

interface ScrambleHoverProps {
  text: string
  className?: string
  as?: keyof JSX.IntrinsicElements
}

export function ScrambleHover({ text, className, as: Tag = "span" }: ScrambleHoverProps) {
  const [displayed, setDisplayed] = useState(text)
  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null)
  const resolvedRef = useRef(0)

  const scramble = () => {
    resolvedRef.current = 0

    // First: full scramble
    intervalRef.current = setInterval(() => {
      setDisplayed(
        text
          .split("")
          .map((char, i) =>
            i < resolvedRef.current
              ? char
              : char === " "
              ? " "
              : CHARS[Math.floor(Math.random() * CHARS.length)]
          )
          .join("")
      )
    }, SCRAMBLE_SPEED)
  }

  const resolve = () => {
    clearInterval(intervalRef.current!)

    const resolveInterval = setInterval(() => {
      resolvedRef.current += 1
      if (resolvedRef.current > text.length) {
        clearInterval(resolveInterval)
        setDisplayed(text)
        return
      }
      setDisplayed(
        text
          .split("")
          .map((char, i) =>
            i < resolvedRef.current
              ? char
              : char === " "
              ? " "
              : CHARS[Math.floor(Math.random() * CHARS.length)]
          )
          .join("")
      )
    }, RESOLVE_SPEED)
  }

  return (
    <Tag
      onMouseEnter={scramble}
      onMouseLeave={resolve}
      className={`cursor-pointer font-mono ${className ?? ""}`}
    >
      {displayed}
    </Tag>
  )
}

CursorFollow

Free Cursor

Custom cursor dot that follows the mouse with spring-based lag. Can expand on hover over interactive elements.

// CursorFollow.tsx
"use client"
import { useEffect, useState } from "react"
import { motion, useSpring } from "framer-motion"

export function CursorFollow() {
  const [visible, setVisible] = useState(false)
  const [hovered, setHovered] = useState(false)

  const cursorX = useSpring(0, { stiffness: 200, damping: 20 })
  const cursorY = useSpring(0, { stiffness: 200, damping: 20 })

  useEffect(() => {
    const move = (e: MouseEvent) => {
      cursorX.set(e.clientX)
      cursorY.set(e.clientY)
      setVisible(true)
    }
    const enter = (e: MouseEvent) => {
      if ((e.target as HTMLElement).closest("a, button")) setHovered(true)
    }
    const leave = (e: MouseEvent) => {
      if ((e.target as HTMLElement).closest("a, button")) setHovered(false)
    }
    window.addEventListener("mousemove", move)
    window.addEventListener("mouseover", enter)
    window.addEventListener("mouseout", leave)
    return () => {
      window.removeEventListener("mousemove", move)
      window.removeEventListener("mouseover", enter)
      window.removeEventListener("mouseout", leave)
    }
  }, [cursorX, cursorY])

  if (!visible) return null

  return (
    <motion.div
      style={{ x: cursorX, y: cursorY, translateX: "-50%", translateY: "-50%" }}
      animate={{ scale: hovered ? 2.5 : 1, opacity: hovered ? 0.5 : 1 }}
      transition={{ scale: { type: "spring", stiffness: 300, damping: 20 } }}
      className="pointer-events-none fixed top-0 left-0 z-[9999] h-4 w-4 rounded-full bg-white mix-blend-difference"
    />
  )
}

Infinite Scroll Marquee

Free Animation Marquee

Infinitely scrolling horizontal ticker using Framer Motion's x-axis animation. Supports pause on hover and reverse direction.

// InfiniteMarquee.tsx
"use client"
import { useRef } from "react"
import { motion, useAnimationFrame, useMotionValue, useTransform, wrap } from "framer-motion"

interface InfiniteMarqueeProps {
  children: React.ReactNode
  speed?: number
  direction?: "left" | "right"
}

export function InfiniteMarquee({ children, speed = 50, direction = "left" }: InfiniteMarqueeProps) {
  const x = useMotionValue(0)
  const containerRef = useRef<HTMLDivElement>(null)

  useAnimationFrame((_, delta) => {
    const moveBy = direction === "left" ? -speed : speed
    x.set(x.get() + (moveBy * delta) / 1000)
  })

  const wrapped = useTransform(x, (v) => {
    const el = containerRef.current
    if (!el) return v
    const width = el.scrollWidth / 2
    return wrap(-width, 0, v)
  })

  return (
    <div className="overflow-hidden">
      <motion.div ref={containerRef} style={{ x: wrapped }} className="flex whitespace-nowrap">
        {/* Render twice for seamless loop */}
        <div className="flex shrink-0">{children}</div>
        <div className="flex shrink-0">{children}</div>
      </motion.div>
    </div>
  )
}

Free Gesture

A horizontal drag slider that requires the user to drag a handle all the way to the right to confirm an action. Common for payment confirmations.

// DragToConfirm.tsx
"use client"
import { useRef, useState } from "react"
import { motion, useMotionValue, useTransform, animate } from "framer-motion"
import { ArrowRight, Check } from "lucide-react"

interface DragToConfirmProps {
  onConfirm: () => void
  label?: string
}

export function DragToConfirm({ onConfirm, label = "Slide to confirm" }: DragToConfirmProps) {
  const trackRef = useRef<HTMLDivElement>(null)
  const x = useMotionValue(0)
  const [confirmed, setConfirmed] = useState(false)
  const HANDLE_SIZE = 48

  const handleWidth = useTransform(x, (v) => v + HANDLE_SIZE)
  const opacity = useTransform(x, [0, 100], [1, 0])

  const handleDragEnd = () => {
    const track = trackRef.current
    if (!track) return
    const maxX = track.offsetWidth - HANDLE_SIZE
    if (x.get() >= maxX * 0.9) {
      animate(x, maxX, { type: "spring", stiffness: 300, damping: 30 })
      setConfirmed(true)
      onConfirm()
    } else {
      animate(x, 0, { type: "spring", stiffness: 300, damping: 30 })
    }
  }

  return (
    <div
      ref={trackRef}
      className="relative h-12 w-full max-w-xs overflow-hidden rounded-full bg-neutral-800"
    >
      {/* Fill background */}
      <motion.div
        style={{ width: handleWidth }}
        className="absolute inset-y-0 left-0 rounded-full bg-indigo-600"
      />
      {/* Label */}
      <motion.span
        style={{ opacity }}
        className="absolute inset-0 flex items-center justify-center text-sm font-medium text-neutral-400 select-none pointer-events-none"
      >
        {label}
      </motion.span>
      {/* Handle */}
      <motion.button
        drag={confirmed ? false : "x"}
        dragConstraints={trackRef}
        dragElastic={0}
        style={{ x }}
        onDragEnd={handleDragEnd}
        className="absolute top-0 left-0 h-12 w-12 rounded-full bg-white shadow-lg flex items-center justify-center cursor-grab active:cursor-grabbing"
      >
        {confirmed ? <Check className="h-5 w-5 text-indigo-600" /> : <ArrowRight className="h-5 w-5 text-indigo-600" />}
      </motion.button>
    </div>
  )
}

  useVelocity,
  wrap,
} from "framer-motion"

interface ScrollVelocityProps {
  text: string
  velocity?: number
  className?: string
}

function VelocityText({ text, velocity = 5, className }: ScrollVelocityProps) {
  const baseX = useMotionValue(0)
  const { scrollY } = useScroll()
  const scrollVelocity = useVelocity(scrollY)
  const smoothVelocity = useSpring(scrollVelocity, { damping: 50, stiffness: 400 })
  const velocityFactor = useTransform(smoothVelocity, [0, 1000], [0, 5], { clamp: false })
  const ref = useRef<HTMLDivElement>(null)
  const x = useTransform(baseX, (v) => {
    const el = ref.current
    if (!el) return v
    return wrap(-el.scrollWidth / 2, 0, v)
  })

  let directionFactor = useRef<number>(1)

  useAnimationFrame((_, delta) => {
    let moveBy = directionFactor.current * velocity * (delta / 1000)
    if (velocityFactor.get() < 0) directionFactor.current = -1
    else if (velocityFactor.get() > 0) directionFactor.current = 1
    moveBy += directionFactor.current * moveBy * velocityFactor.get()
    baseX.set(baseX.get() + moveBy)
  })

  return (
    <div className="overflow-hidden">
      <motion.div ref={ref} style={{ x }} className={`flex whitespace-nowrap ${className ?? ""}`}>
        {Array(4).fill(text).map((t, i) => (
          <span key={i} className="mr-12 text-4xl font-black uppercase">{t}</span>
        ))}
      </motion.div>
    </div>
  )
}

export function ScrollVelocity({ text, velocity, className }: ScrollVelocityProps) {
  return (
    <section className="py-8 space-y-4">
      <VelocityText text={text} velocity={velocity} className={className} />
      <VelocityText text={text} velocity={-(velocity ?? 5)} className={className} />
    </section>
  )
}
