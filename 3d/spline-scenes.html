Spline 3D Scenes

Production-ready interactive 3D for web hero sections — no WebGL expertise required. Spline outputs optimized .splinecode bundles embeddable via a web component or React package.

@splinetool/react-spline @splinetool/runtime @splinetool/viewer React Next.js Free tier

Overview

Spline is a browser-based 3D design tool that exports interactive scenes directly to the web. Scenes are rendered via WebGL at runtime — no build pipeline required for embedding. The community library has hundreds of free remixable scenes.

No Three.js knowledge

Visual editor outputs ready-to-use scene URLs

Real-time interactivity

Mouse follow, scroll triggers, click events built in

Community scenes

Hundreds of free remixable hero scenes

Installation

# npm
npm install @splinetool/react-spline @splinetool/runtime

# yarn
yarn add @splinetool/react-spline @splinetool/runtime

The @splinetool/runtime peer dependency is required — it provides the core WebGL renderer. Both packages must be installed together.

export default function Interactive() {
  function onMouseDown(e) {
    if (e.target.name === 'Button') {
      console.log('3D button clicked!', e.target);
    }
  }

  function onMouseHover(e) {
    console.log('Hovering over:', e.target.name);
  }

  return (
    <Spline
      scene="https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode"
      onSplineMouseDown={onMouseDown}

Emitting events from outside the scene

import { useRef } from 'react';
import Spline from '@splinetool/react-spline';

export default function Hero() {
  const splineApp = useRef(null);

  function onLoad(spline) {
    splineApp.current = spline;
  }

  // Trigger the 'mouseHover' animation on the 'Orb' object
  function triggerOrbHover() {
    splineApp.current?.emitEvent('mouseHover', 'Orb');
  }

  // Reverse it (like mouse leave)
  function releaseOrb() {
    splineApp.current?.emitEventReverse('mouseHover', 'Orb');
  }

  return (
    <>
      <Spline
        scene="https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode"
        onLoad={onLoad}
      />
      <button onMouseEnter={triggerOrbHover} onMouseLeave={releaseOrb}>
        Hover to animate
      </button>
    </>
  );
}

Mouse-follow (cursor tracking via variables)

In the Spline editor: Events → Mouse Move → Look At / Follow Mouse handles cursor tracking without JS. For custom sensitivity, use scene variables:

import { useRef, useEffect } from 'react';
import Spline from '@splinetool/react-spline';

export default function MouseFollow() {
  const splineApp = useRef(null);

  useEffect(() => {
    const handleMouseMove = (e) => {
      if (!splineApp.current) return;
      // Normalize to -1..1 range
      const x = (e.clientX / window.innerWidth) * 2 - 1;
      const y = -(e.clientY / window.innerHeight) * 2 + 1;
      // Pass to variables you've defined in the Spline editor
      splineApp.current.setVariable?.('mouseX', x);
      splineApp.current.setVariable?.('mouseY', y);
    };
    window.addEventListener('mousemove', handleMouseMove, { passive: true });
    return () => window.removeEventListener('mousemove', handleMouseMove);
  }, []);

  return (
    <Spline
      scene="https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode"
      onLoad={(app) => { splineApp.current = app; }}
    />
  );
}

Scroll-driven animation via variables

import { useRef, useEffect } from 'react';
import Spline from '@splinetool/react-spline';

export default function ScrollScene() {
  const splineApp = useRef(null);

  useEffect(() => {
    const handleScroll = () => {
      if (!splineApp.current) return;
      const progress = window.scrollY / (document.body.scrollHeight - window.innerHeight);
      // Pass 0-1 scroll progress into Spline
      splineApp.current.setVariable?.('scrollProgress', progress);
      // Or emit a discrete event at a threshold
      if (progress > 0.5) {
        splineApp.current.emitEvent('scroll', 'Timeline');
      }
    };
    window.addEventListener('scroll', handleScroll, { passive: true });
    return () => window.removeEventListener('scroll', handleScroll);
  }, []);

  return (
    <Spline
      scene="https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode"
      onLoad={(app) => { splineApp.current = app; }}
      renderOnDemand={false}
    />
  );
}

Object Control & Scene Manipulation

import { useRef } from 'react';
import Spline from '@splinetool/react-spline';

export default function SceneControl() {
  const splineApp = useRef(null);
  const cubeRef = useRef(null);

  function onLoad(spline) {
    splineApp.current = spline;
    // Cache object reference on load — cheaper than finding it each time
    cubeRef.current = spline.findObjectByName('Cube');
  }

  function moveRight() {
    if (cubeRef.current) cubeRef.current.position.x += 10;
  }

  function rotateY() {
    if (cubeRef.current) cubeRef.current.rotation.y += Math.PI / 4;
  }

  function zoomIn() {
    splineApp.current?.setZoom(1.5);
  }

  return (
    <>
      <Spline scene="..." onLoad={onLoad} />
      <button onClick={moveRight}>Move</button>
      <button onClick={rotateY}>Rotate</button>
      <button onClick={zoomIn}>Zoom</button>
    </>
  );
}

Mobile: skip Spline, show static fallback

'use client';
import { useEffect, useState, Suspense, lazy } from 'react';

const Spline = lazy(() => import('@splinetool/react-spline'));

export default function AdaptiveHero() {
  const [isMobile, setIsMobile] = useState(false);

  useEffect(() => {
    const mq = window.matchMedia('(max-width: 768px)');
    setIsMobile(mq.matches);
    const handler = (e) => setIsMobile(e.matches);
    mq.addEventListener('change', handler);
    return () => mq.removeEventListener('change', handler);
  }, []);

  // On mobile: static gradient saves ~170KB and GPU load
  if (isMobile) {
    return <div style={{ height: '60vh', background: 'linear-gradient(135deg, #0a0a0a, #1a0a2e)' }} />;
  }

  return (
    <Suspense fallback={<div style={{ height: '100vh', background: '#0a0a0a' }} />}>
      <Spline scene="https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode" />
    </Suspense>
  );
}

Performance Optimization

1. renderOnDemand (default: true) — most impactful

When true, Spline only re-renders the WebGL canvas when user interaction occurs. Saves significant GPU time on idle frames. Keep this true unless you have continuously looping animations.

// Keep true (default) — idle scenes save GPU
<Spline scene="..." renderOnDemand={true} />

// Set false only for continuously animating scenes (particles loop, timeline plays)
<Spline scene="..." renderOnDemand={false} />

2. Always lazy-load — never eager import

// Bad — blocks main bundle parse
import Spline from '@splinetool/react-spline';

// Good — splits into separate async chunk
const Spline = React.lazy(() => import('@splinetool/react-spline'));

// Next.js equivalent:
const Spline = dynamic(() => import('@splinetool/react-spline'), { ssr: false });

3. Preload the scene file

<!-- In Next.js app/layout.tsx or <head> -->
<link
  rel="preload"
  href="https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode"
  as="fetch"
  crossOrigin="anonymous"
/>

4. Intersection Observer — mount only when visible

'use client';
import { useRef, useState, useEffect, Suspense, lazy } from 'react';

const Spline = lazy(() => import('@splinetool/react-spline'));

export default function LazyScene() {
  const containerRef = useRef(null);
  const [inView, setInView] = useState(false);

  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => { if (entry.isIntersecting) setInView(true); },
      { threshold: 0.1 }
    );
    if (containerRef.current) observer.observe(containerRef.current);
    return () => observer.disconnect();
  }, []);

  return (
    <div ref={containerRef} style={{ height: '100vh' }}>
      {inView && (
        <Suspense fallback={null}>
          <Spline scene="https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode" />
        </Suspense>
      )}
    </div>
  );
}

5. Scene optimization in the Spline editor

In-editor checklist:
  • Reduce polygon count — use Spline's mesh simplification tool
  • Compress textures — keep hero scene textures under 1024×1024
  • Keep particle count under 10,000 for 60fps on mid-range GPUs
  • Use the export quality slider — Medium is the right balance
  • Disable post-processing (bloom, DOF) unless scene centerpiece
  • Merge static geometry — fewer draw calls = better performance
  • Use baked lighting instead of real-time shadow-casting lights

GSAP Integration

GSAP controls the page layer (DOM, CSS). Spline controls the 3D layer. Bridge them via the onLoad callback and direct SplineObject property mutation.

Entrance animation — GSAP timeline triggers Spline event

'use client';
import { useRef } from 'react';
import Spline from '@splinetool/react-spline/next';
import gsap from 'gsap';

export default function GSAPHero() {
  const headlineRef = useRef(null);
  const ctaRef = useRef(null);
  const splineApp = useRef(null);

  function onSplineLoad(spline) {
    splineApp.current = spline;

    // Chain: DOM animates in, then triggers 3D entrance
    const tl = gsap.timeline({ defaults: { ease: 'power3.out' } });
    tl.from(headlineRef.current, { y: 60, opacity: 0, duration: 1 })
      .from(ctaRef.current, { y: 30, opacity: 0, duration: 0.7 }, '-=0.4')
      .call(() => {
        // Fire the 3D entrance animation after DOM settles
        spline.emitEvent('start', 'HeroObject');
      }, null, '-=0.2');
  }

  return (
    <section style={{ position: 'relative', height: '100vh', overflow: 'hidden' }}>
      <Spline
        scene="https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode"
        onLoad={onSplineLoad}
        style={{ position: 'absolute', inset: 0 }}
      />
      <div style={{ position: 'relative', zIndex: 10, padding: '20vh 8vw' }}>
        <h1 ref={headlineRef}>Build in 3D</h1>
        <button ref={ctaRef}>Get started</button>
      </div>
    </section>
  );
}

ScrollTrigger → drive Spline object rotation

'use client';
import { useRef } from 'react';
import Spline from '@splinetool/react-spline/next';
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';

gsap.registerPlugin(ScrollTrigger);

export default function ScrollScene() {
  const splineApp = useRef(null);
  const sphereRef = useRef(null);

  function onLoad(spline) {
    splineApp.current = spline;
    sphereRef.current = spline.findObjectByName('Sphere');

    if (!sphereRef.current) return;

    // GSAP tweens SplineObject properties directly
    gsap.to(sphereRef.current.rotation, {
      y: Math.PI * 2,
      ease: 'none',
      scrollTrigger: {
        trigger: 'body',
        start: 'top top',
        end: 'bottom bottom',
        scrub: 1,
      },
    });

    gsap.to(sphereRef.current.position, {
      z: -300,
      ease: 'none',
      scrollTrigger: {
        trigger: 'body',
        start: '30% top',
        end: '70% top',
        scrub: 1.5,
      },
    });
  }

  return (
    <div style={{ height: '300vh' }}>
      <div style={{ position: 'sticky', top: 0, height: '100vh' }}>
        <Spline
          scene="https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode"
          onLoad={onLoad}
          renderOnDemand={false} {/* required for scrub animations */}
        />
      </div>
    </div>
  );
}
Important: Set renderOnDemand={false} when using GSAP scrub or ticker-driven animations. With renderOnDemand: true, the canvas won't update between GSAP ticks since there's no user interaction event to trigger a re-render.

GSAP ticker → continuous Spline update loop

import { useRef, useEffect } from 'react';
import Spline from '@splinetool/react-spline/next';
import gsap from 'gsap';

export default function TickerScene() {
  const splineApp = useRef(null);
  const objRef = useRef(null);

  function onLoad(spline) {
    splineApp.current = spline;
    objRef.current = spline.findObjectByName('Ring');
  }

  useEffect(() => {
    const tick = () => {
      if (!objRef.current) return;
      objRef.current.rotation.y += 0.005;
      // Force re-render each tick
      splineApp.current?.requestRender?.();
    };
    gsap.ticker.add(tick);
    return () => gsap.ticker.remove(tick);
  }, []);

  return <Spline scene="..." onLoad={onLoad} renderOnDemand={false} />;
}

Framer Motion Integration

Fade-in with loading skeleton using AnimatePresence

'use client';
import { useState } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import Spline from '@splinetool/react-spline/next';

export default function FramerSplineHero() {
  const [loaded, setLoaded] = useState(false);

  return (
    <section style={{ position: 'relative', height: '100vh' }}>
      {/* Skeleton that fades out when scene loads */}
      <AnimatePresence>
        {!loaded && (
          <motion.div
            key="skeleton"
            initial={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            transition={{ duration: 0.8 }}
            style={{
              position: 'absolute', inset: 0,
              background: 'linear-gradient(135deg, #0a0a0a 0%, #0f0f1a 100%)',
            }}
          />
        )}
      </AnimatePresence>

      {/* Scene fades in on load */}
      <motion.div
        initial={{ opacity: 0 }}
        animate={{ opacity: loaded ? 1 : 0 }}
        transition={{ duration: 1.2, ease: 'easeOut' }}
        style={{ height: '100%' }}
      >
        <Spline
          scene="https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode"
          onLoad={() => setLoaded(true)}
        />
      </motion.div>

      {/* Content animates in after scene is ready */}
      <motion.div
        initial={{ y: 40, opacity: 0 }}
        animate={{ y: loaded ? 0 : 40, opacity: loaded ? 1 : 0 }}
        transition={{ duration: 0.9, delay: 0.4, ease: [0.16, 1, 0.3, 1] }}
        style={{ position: 'absolute', bottom: '10vh', left: '8vw', zIndex: 10 }}
      >
        <h1>Your headline</h1>
      </motion.div>
    </section>
  );
}

useScroll + useMotionValueEvent → drive Spline properties

'use client';
import { useRef } from 'react';
import { useScroll, useMotionValueEvent } from 'framer-motion';
import Spline from '@splinetool/react-spline/next';

export default function FramerScrollSpline() {
  const containerRef = useRef(null);
  const splineApp = useRef(null);
  const objRef = useRef(null);

  const { scrollYProgress } = useScroll({
    target: containerRef,
    offset: ['start start', 'end end'],
  });

  useMotionValueEvent(scrollYProgress, 'change', (progress) => {
    if (!objRef.current) return;
    objRef.current.rotation.y = progress * Math.PI * 2;
    objRef.current.scale.setScalar(1 + progress * 0.5);
    splineApp.current?.requestRender?.();
  });

  function onLoad(spline) {
    splineApp.current = spline;
    objRef.current = spline.findObjectByName('HeroOrb');
  }

  return (
    <div ref={containerRef} style={{ height: '300vh' }}>
      <div style={{ position: 'sticky', top: 0, height: '100vh' }}>
        <Spline
          scene="https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode"
          onLoad={onLoad}
          renderOnDemand={false}
        />
      </div>
    </div>
  );
}

Vanilla JS / @splinetool/runtime

For non-React frameworks (Vue, Svelte, Astro, plain HTML) or direct canvas control without the React wrapper.

import { Application } from '@splinetool/runtime';

const canvas = document.getElementById('spline-canvas');
const spline = new Application(canvas);

await spline.load('https://prod.spline.design/YOUR_SCENE_ID/scene.splinecode');

obj.position.x += 50;
spline.emitEvent('mouseHover', 'Hero');
spline.setZoom(1.2);

Workflow: community scene to production

1. community.spline.design → find a scene → click "Remix"
2. Opens in your Spline editor account
3. Customize: colors, materials, timing, particle count
4. File → Publish → copy the prod.spline.design URL
   OR: Export → For Web → Download .splinecode → /public/scenes/
5. Embed with <Spline scene="..." />
6. Chrome DevTools → Performance tab → profile, check GPU usage
7. Tune: renderOnDemand, scene quality, lazy loading threshold

Watermark

Info: The Spline watermark appears on scenes hosted on Spline's CDN under the free plan. To remove it: upgrade to a paid Spline plan, or self-host the .splinecode file from your own domain (the watermark is injected by the Spline CDN layer, not the viewer library).

Scene URL format reference
