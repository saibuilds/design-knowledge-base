React Three Fiber

React renderer for Three.js. Write declarative 3D scenes using JSX components. Pairs with @react-three/drei for helpers, @react-three/postprocessing for visual effects, and @react-three/rapier for physics simulation.

Installation

bash
npm install three @react-three/fiber @react-three/drei
npm install @react-three/postprocessing   # bloom, DOF, etc.
npm install @react-three/rapier           # physics
npm install @types/three                  # TypeScript
Next.js App Router: Three.js needs browser APIs. Wrap in dynamic import with ssr:false, or mark as 'use client'. The Canvas element must render client-side only.
tsx — Next.js safe import
import dynamic from 'next/dynamic';

const Scene = dynamic(() => import('@/components/three/Scene'), { ssr: false });

export default function Page() {
  return (
    <main>
      <Scene />
    </main>
  );
}

Canvas and Scene Setup

tsx
'use client';
import { Canvas } from '@react-three/fiber';
import { OrbitControls, Environment } from '@react-three/drei';
import { Suspense } from 'react';

export default function Scene() {
  return (
    <Canvas
      style={{ width: '100%', height: '100vh' }}
      camera={{ position: [0, 0, 5], fov: 75 }}
      gl={{ antialias: true, alpha: true }}   // alpha: transparent bg
      shadows                                  // enable shadow maps
      dpr={[1, 2]}                            // pixel ratio range
      onCreated={({ gl }) => gl.setClearColor('#000000', 0)}
    >
      <Suspense fallback={null}>
        <ambientLight intensity={0.5} />
        <directionalLight position={[5, 5, 5]} intensity={1} castShadow />

        <mesh>
          <boxGeometry args={[1, 1, 1]} />
          <meshStandardMaterial color="orange" />
        </mesh>

        <OrbitControls enableZoom enablePan enableRotate />
        <Environment preset="city" />
      </Suspense>
    </Canvas>
  );
}

Meshes, Geometries, Materials

tsx
// Mesh = geometry + material
<mesh position={[0, 0, 0]} rotation={[0, Math.PI / 4, 0]} scale={1.5}>
  <boxGeometry args={[1, 1, 1]} />
  <meshStandardMaterial color="#f59e0b" metalness={0.5} roughness={0.3} />
</mesh>

// Geometries
<sphereGeometry args={[1, 32, 32]} />           // radius, wSeg, hSeg
<cylinderGeometry args={[0.5, 0.5, 2, 32]} />  // topR, bottomR, height, seg
<torusGeometry args={[1, 0.3, 16, 100]} />     // radius, tube, rSeg, tSeg
<planeGeometry args={[5, 5, 50, 50]} />         // w, h, wSeg, hSeg
<icosahedronGeometry args={[1, 4]} />           // radius, detail

// Materials
<meshBasicMaterial color="red" />              // no lighting needed
<meshStandardMaterial color="blue" metalness={0.8} roughness={0.2} />
<meshPhysicalMaterial                          // glass
  color="white" transmission={0.9} thickness={0.5} roughness={0.05} ior={1.5}
/>
<meshNormalMaterial />                         // debug normals
<pointsMaterial size={0.05} color="white" />   // particles

// Textured
import { useTexture } from '@react-three/drei';
function TexturedMesh() {
  const map = useTexture('/textures/wood.jpg');
  return <mesh><boxGeometry /><meshStandardMaterial map={map} /></mesh>;
}

useFrame Animation Loop

tsx
import { useFrame } from '@react-three/fiber';
import { useRef } from 'react';
import type { Mesh } from 'three';

function RotatingCube() {
  const meshRef = useRef<Mesh>(null);

  useFrame((state, delta) => {
    // delta = seconds since last frame (~0.016 at 60fps)
    if (!meshRef.current) return;
    meshRef.current.rotation.y += delta * 0.8;
    // Breathing scale
    meshRef.current.scale.setScalar(1 + Math.sin(state.clock.elapsedTime * 2) * 0.05);
  });

  return (
    <mesh ref={meshRef}>
      <boxGeometry />
      <meshStandardMaterial color="orange" />
    </mesh>
  );
}

// Mouse follow
function MouseFollower() {
  const meshRef = useRef<Mesh>(null);
  useFrame(({ pointer }) => {
    if (!meshRef.current) return;
    meshRef.current.position.x += (pointer.x * 3 - meshRef.current.position.x) * 0.05;
    meshRef.current.position.y += (pointer.y * 3 - meshRef.current.position.y) * 0.05;
  });
  return <mesh ref={meshRef}><sphereGeometry args={[0.3]} /><meshNormalMaterial /></mesh>;
}

// quickTo for smooth lerp (pair with GSAP)
import gsap from 'gsap';
const moveX = gsap.quickTo(obj, 'x', { duration: 0.3 });
useFrame(({ pointer }) => moveX(pointer.x * 3));

Drei Helpers Reference

HelperDescription
OrbitControlsMouse orbit, zoom, pan. Add autoRotate for auto-spin.
EnvironmentHDR environment lighting. Presets: city, studio, sunset, forest, apartment, dawn, night, park, lobby, warehouse
ContactShadowsFake (cheap) shadow blob on ground plane — no real shadow maps needed
FloatGentle floating/bobbing animation. Props: speed, rotationIntensity, floatIntensity
Text3D text via troika-three-text. Props: font, fontSize, color, anchorX/Y
HtmlHTML rendered at a 3D position. Props: center, distanceFactor, occlude
useGLTFLoad and cache .glb models with Draco decompression
useTextureLoad and cache textures. Pass string or object of {map, normalMap, etc.}
PresentationControlsTouch-friendly drag rotation, no scroll hijack — ideal for product viewers
StageComplete product demo setup: lighting + environment + shadows in one component
CenterAuto-centers geometry/model. Crucial for imported models with off-center origins.
BoundsFits camera to scene content with padding
ScrollControlsScroll-linked scenes within R3F. Alternative to GSAP for pure 3D scroll.
StarsParticle star field. Props: radius, depth, count, factor, saturation, fade
SparklesSparkle particles. Props: count, scale, size, speed, color, noise
MeshReflectorMaterialMirror/wet floor reflection material
MeshTransmissionMaterialHigh-quality glass/crystal material
SoftShadowsPCF soft shadow sampling — add inside Canvas
LoaderProgress bar UI for Suspense loading state
tsx — drei usage
import {
  OrbitControls, Environment, Float, Text, Html,
  useGLTF, ContactShadows, Stage, Stars, Sparkles,
  PresentationControls
} from '@react-three/drei';

// GLTF model
function Model() {
  const { scene } = useGLTF('/product.glb');
  return <primitive object={scene} />;
}
useGLTF.preload('/product.glb'); // preload before component renders

// Floating wrapper
<Float speed={2} rotationIntensity={0.5} floatIntensity={1}>
  <Model />
</Float>

// HTML label attached to 3D position
<Html position={[0, 1.5, 0]} center distanceFactor={5}>
  <div style={{ background: 'black', color: 'white', padding: '4px 8px', borderRadius: '4px' }}>
    Label
  </div>
</Html>

// Product demo — one-liner setup
<Stage environment="city" intensity={0.6} shadows="contact">
  <Center><Model /></Center>
</Stage>

// Star field + sparkles
<Stars radius={100} depth={50} count={5000} factor={4} saturation={0} fade speed={1} />
<Sparkles count={200} scale={4} size={2} speed={0.4} color="gold" />

Lighting

tsx
// Ambient — uniform fill (no direction, no shadows)
<ambientLight intensity={0.3} />

// Directional — sun-like parallel rays
<directionalLight
  position={[5, 10, 5]}
  intensity={1.5}
  castShadow
  shadow-mapSize={[2048, 2048]}
  shadow-camera-far={50}
/>

// Point — omnidirectional bulb
<pointLight position={[0, 3, 0]} intensity={2} color="#ff9900" decay={2} />

// Spot — cone of light
<spotLight
  position={[0, 10, 0]}
  angle={Math.PI / 6}
  penumbra={0.3}    // soft edge 0-1
  intensity={2}
  castShadow
/>

// Best practice for product renders
<ambientLight intensity={0.2} />
<Environment preset="studio" />
<directionalLight position={[5, 8, 3]} intensity={0.8} castShadow />

Custom Shaders

tsx
import { shaderMaterial } from '@react-three/drei';
import { extend, useFrame } from '@react-three/fiber';
import { useRef } from 'react';
import * as THREE from 'three';

const WaveMaterial = shaderMaterial(
  { uTime: 0, uColor: new THREE.Color(0.2, 0.5, 1.0) },
  // Vertex shader
  `uniform float uTime; varying vec2 vUv;
   void main() {
     vUv = uv;
     vec3 pos = position;
     pos.z += sin(pos.x * 3.0 + uTime * 2.0) * 0.15;
     gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);
   }`,
  // Fragment shader
  `uniform vec3 uColor; uniform float uTime; varying vec2 vUv;
   void main() {
     float wave = 0.5 + 0.5 * sin(vUv.x * 10.0 + uTime);
     gl_FragColor = vec4(uColor * wave, 1.0);
   }`
);

extend({ WaveMaterial });

function WavePlane() {
  const matRef = useRef<any>(null);
  useFrame(({ clock }) => { matRef.current.uTime = clock.elapsedTime; });
  return (
    <mesh rotation={[-Math.PI / 2, 0, 0]}>
      <planeGeometry args={[5, 5, 64, 64]} />
      <waveMaterial ref={matRef} side={THREE.DoubleSide} />
    </mesh>
  );
}

Production Pattern — Scroll-Animated Product Viewer

tsx — complete scene
'use client';
import { Canvas, useFrame } from '@react-three/fiber';
import { OrbitControls, Environment, Float, useGLTF, Center, Loader, ContactShadows } from '@react-three/drei';
import { EffectComposer, Bloom } from '@react-three/postprocessing';
import { Suspense, useRef } from 'react';
import type { Group } from 'three';

function Model({ scrollProgress = 0 }) {
  const { scene } = useGLTF('/product.glb');
  const groupRef = useRef<Group>(null);

  useFrame(({ clock }) => {
    if (!groupRef.current) return;
    groupRef.current.rotation.y = scrollProgress * Math.PI * 2;
    groupRef.current.position.y = Math.sin(clock.elapsedTime * 0.5) * 0.05;
  });

  return (
    <group ref={groupRef}>
      <Center><primitive object={scene} /></Center>
    </group>
  );
}
useGLTF.preload('/product.glb');

export default function ProductViewer({ scrollProgress = 0 }: { scrollProgress?: number }) {
  return (
    <div style={{ width: '100%', height: '100vh', position: 'relative' }}>
      <Canvas
        camera={{ position: [0, 0.5, 4], fov: 40 }}
        dpr={[1, 2]}
        gl={{ antialias: true, alpha: true }}
        shadows
      >
        <Suspense fallback={null}>
          <ambientLight intensity={0.2} />
          <Environment preset="studio" />
          <Float speed={1.5} rotationIntensity={0.2}>
            <Model scrollProgress={scrollProgress} />
          </Float>
          <ContactShadows position={[0, -1.5, 0]} opacity={0.4} blur={2.5} far={4} />
          <EffectComposer>
            <Bloom intensity={0.25} luminanceThreshold={0.9} mipmapBlur />
          </EffectComposer>
        </Suspense>
        <OrbitControls enableZoom={false} enablePan={false} autoRotate autoRotateSpeed={0.3} />
      </Canvas>
      <Loader />
    </div>
  );
}
