Shader Basics

ShaderMaterial in Three.js
const material = new THREE.ShaderMaterial({
  uniforms: {
    uTime:  { value: 0 },
    uColor: { value: new THREE.Color('#f59e0b') }
  },
  vertexShader: /* glsl */`
    uniform float uTime;
    varying vec2 vUv;

    void main() {
      vUv = uv;
      vec3 pos = position;
      pos.z += sin(pos.x * 3.0 + uTime) * 0.1;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);
    }
  `,
  fragmentShader: /* glsl */`
    uniform vec3 uColor;
    varying vec2 vUv;

    void main() {
      gl_FragColor = vec4(uColor, 1.0);
    }
  `
});

// In animation loop — update time uniform
function animate() {
  requestAnimationFrame(animate);
  material.uniforms.uTime.value += 0.01;
  renderer.render(scene, camera);
}

Common Fragment Shader Patterns

Gradient by UV
// Fragment shader
varying vec2 vUv;

void main() {
  // Vertical gradient: black to amber
  vec3 colorA = vec3(0.0, 0.0, 0.0);
  vec3 colorB = vec3(0.96, 0.62, 0.04);  // #f59e0b
  vec3 color = mix(colorA, colorB, vUv.y);
  gl_FragColor = vec4(color, 1.0);
}
Animated noise distortion
// Vertex shader with animated wave distortion
uniform float uTime;
uniform float uStrength;
varying vec2 vUv;

// Simple noise
float noise(vec2 p) {
  return sin(p.x * 4.0) * sin(p.y * 4.0);
}

void main() {
  vUv = uv;
  vec3 pos = position;
  float n = noise(uv + uTime * 0.3);
  pos.z += n * uStrength;
  gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);
}

Using Shaders in R3F

shaderMaterial with @react-three/drei
import { shaderMaterial } from '@react-three/drei';
import { extend, useFrame } from '@react-three/fiber';
import { useRef } from 'react';
import * as THREE from 'three';

const WaveMaterial = shaderMaterial(
  { uTime: 0, uColor: new THREE.Color('#f59e0b') },
  // Vertex
  `uniform float uTime;
   varying vec2 vUv;
   void main() {
     vUv = uv;
     vec3 pos = position;
     pos.z += sin(pos.x * 3.0 + uTime) * 0.05;
     gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);
   }`,
  // Fragment
  `uniform vec3 uColor;
   void main() {
     gl_FragColor = vec4(uColor, 1.0);
   }`
);

extend({ WaveMaterial });

function WavePlane() {
  const matRef = useRef<any>(null);

  useFrame(({ clock }) => {
    if (matRef.current) matRef.current.uTime = clock.elapsedTime;
  });

  return (
    <mesh>
      <planeGeometry args={[3, 3, 64, 64]} />
      <waveMaterial ref={matRef} />
    </mesh>
  );
}
