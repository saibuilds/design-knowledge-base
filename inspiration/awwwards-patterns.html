Hero Patterns

Kinetic typography — scramble text
Characters scramble to reveal final text on load. Seen on tech/agency sites.
JS
function scrambleText(el, finalText, duration = 1200) {
  const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789!@#$%';
  const steps = 20;
  const interval = duration / steps;
  let step = 0;

  const timer = setInterval(() => {
    el.textContent = finalText
      .split('')
      .map((char, i) => {
        if (char === ' ') return ' ';
        if (i < (step / steps) * finalText.length) return char;
        return chars[Math.floor(Math.random() * chars.length)];
      })
      .join('');

    if (++step > steps) {
      clearInterval(timer);
      el.textContent = finalText;
    }
  }, interval);
}

// Usage
scrambleText(document.querySelector('h1'), 'DESIGN STUDIO', 1500);

3D object hero with R3F
WebGL canvas behind hero text, object rotates on mouse move.
R3F
'use client';
import { Canvas, useFrame } from '@react-three/fiber';
import { Environment, Float } from '@react-three/drei';
import { useRef } from 'react';

function HeroObject() {
  const ref = useRef(null);
  useFrame((state) => {
    if (!ref.current) return;
    ref.current.rotation.y = state.mouse.x * 0.4;
    ref.current.rotation.x = -state.mouse.y * 0.2;
  });
  return (
    <Float speed={2} rotationIntensity={0.3} floatIntensity={0.5}>
      <mesh ref={ref}>
        <torusKnotGeometry args={[0.8, 0.25, 256, 32]} />
        <meshStandardMaterial color="#f59e0b" metalness={0.8} roughness={0.1} />
      </mesh>
    </Float>
  );
}

export default function Hero3D() {
  return (
    <section style={{ position: 'relative', height: '100vh' }}>
      <div style={{ position: 'absolute', inset: 0 }}>
        <Canvas camera={{ position: [0, 0, 4] }} gl={{ alpha: true }}>
          <ambientLight intensity={0.5} />
          <Environment preset="city" />
          <HeroObject />
        </Canvas>
      </div>
      <div style={{ position: 'relative', zIndex: 10 }}>
        <h1>Your Studio Name</h1>
      </div>
    </section>
  );
}

Scroll Patterns

Pinned horizontal scroll section
Section pins while user scrolls, content moves horizontally. Used for project showcases.
GSAP + ScrollTrigger
const track = document.querySelector('.h-track');
const panels = gsap.utils.toArray('.h-panel');

gsap.to(panels, {
  xPercent: -100 * (panels.length - 1),
  ease: 'none',
  scrollTrigger: {
    trigger: '.h-scroll-section',
    pin: true,
    scrub: 1,
    snap: 1 / (panels.length - 1),
    end: () => '+=' + track.offsetWidth
  }
});

Interaction Patterns

Custom magnetic cursor
Cursor dot that lags behind and scales up on hover. Standard on creative agencies.
GSAP
const cursor = document.querySelector('.cursor');
const follower = document.querySelector('.cursor-follower');

let mouseX = 0, mouseY = 0;
let followerX = 0, followerY = 0;

window.addEventListener('mousemove', (e) => {
  mouseX = e.clientX;
  mouseY = e.clientY;
  gsap.to(cursor, { x: mouseX, y: mouseY, duration: 0.1 });
});

// Smooth follower
gsap.ticker.add(() => {
  followerX += (mouseX - followerX) * 0.1;
  followerY += (mouseY - followerY) * 0.1;
  gsap.set(follower, { x: followerX, y: followerY });
});

// Hover effect on interactive elements
document.querySelectorAll('a, button').forEach(el => {
  el.addEventListener('mouseenter', () => {
    gsap.to(cursor, { scale: 3, duration: 0.3 });
  });
  el.addEventListener('mouseleave', () => {
    gsap.to(cursor, { scale: 1, duration: 0.3 });
  });
});

Page transition wipe
Full-screen overlay slides in/out between route changes. With Framer Motion in Next.js.
Framer Motion
// components/PageTransition.tsx
'use client';
import { motion, AnimatePresence } from 'framer-motion';
import { usePathname } from 'next/navigation';

export function PageTransition({ children }: { children: React.ReactNode }) {
  const pathname = usePathname();

  return (
    <AnimatePresence mode="wait">
      <motion.div key={pathname}>
        {/* Wipe overlay */}
        <motion.div
          className="fixed inset-0 bg-black z-50 pointer-events-none"
          initial={{ scaleY: 0, transformOrigin: 'top' }}
          animate={{ scaleY: 0, transformOrigin: 'bottom' }}
          exit={{ scaleY: 1, transformOrigin: 'top' }}
          transition={{ duration: 0.5, ease: [0.76, 0, 0.24, 1] }}
        />
        {/* Page content */}
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1, transition: { delay: 0.3 } }}
          exit={{ opacity: 0 }}
        >
          {children}
        </motion.div>
      </motion.div>
    </AnimatePresence>
  );
}
