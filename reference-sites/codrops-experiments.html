Text Scramble — Standalone Pattern

// Text scramble — Codrops-style implementation
class TextScramble {
  constructor(el) {
    this.el = el
    this.chars = '!<>-_\\/[]{}—=+*^?#__abcdefghijklmnopqrstuvwxyz'
    this.update = this.update.bind(this)
  }

  setText(newText) {
    const oldText = this.el.innerText
    const length = Math.max(oldText.length, newText.length)
    const promise = new Promise(resolve => (this.resolve = resolve))
    this.queue = []

    for (let i = 0; i < length; i++) {
      const from = oldText[i] || ''
      const to   = newText[i] || ''
      const start = Math.floor(Math.random() * 20)
      const end   = start + Math.floor(Math.random() * 20)
      this.queue.push({ from, to, start, end })
    }

    cancelAnimationFrame(this.frameRequest)
    this.frame = 0
    this.update()
    return promise
  }

  update() {
    let output = ''
    let complete = 0

    for (let i = 0, n = this.queue.length; i < n; i++) {
      let { from, to, start, end, char } = this.queue[i]

      if (this.frame >= end) {
        complete++
        output += to
      } else if (this.frame >= start) {
        if (!char || Math.random() < 0.28) {
          char = this.chars[Math.floor(Math.random() * this.chars.length)]
          this.queue[i].char = char
        }
        output += `<span style="color:#f59e0b">${char}</span>`
      } else {
        output += from
      }
    }

    this.el.innerHTML = output
    if (complete === this.queue.length) {
      this.resolve()
    } else {
      this.frameRequest = requestAnimationFrame(this.update)
      this.frame++
    }
  }
}

// Usage:
const el = document.querySelector('.scramble-text')
const fx = new TextScramble(el)
const phrases = ['Creative Developer', 'Design Engineer', 'Frontend Artisan']
let counter = 0
const cycle = () => {
  fx.setText(phrases[counter]).then(() => {
    setTimeout(cycle, 2000)
  })
  counter = (counter + 1) % phrases.length
}
cycle()

Page Transition with Clip-Path

// Codrops page transition pattern — overlay wipe
import gsap from 'gsap'

const overlay = document.querySelector('.page-overlay')

// Enter: wipe in (overlay covers page)
async function pageOut(href) {
  await gsap.to(overlay, {
    clipPath: 'inset(0 0% 0 0)',  // fully visible
    duration: 0.6,
    ease: 'power2.inOut',
  })
  window.location.href = href
}

// Exit: wipe out (overlay reveals new page)
window.addEventListener('DOMContentLoaded', () => {
  gsap.fromTo(overlay,
    { clipPath: 'inset(0 0% 0 0)' },
    { clipPath: 'inset(0 100% 0 0)', duration: 0.6, ease: 'power2.inOut', delay: 0.1 }
  )
})

document.querySelectorAll('a[data-transition]').forEach(link => {
  link.addEventListener('click', (e) => {
    e.preventDefault()
    pageOut(link.href)
  })
})

/* Overlay element */
.page-overlay {
  position: fixed;
  inset: 0;
  background: #0a0a0a;
  z-index: 9999;
  pointer-events: none;
  clip-path: inset(0 100% 0 0);
}

Cursor Ripple on Canvas

// Codrops-style canvas ripple cursor effect
const canvas = document.querySelector('#cursor-canvas')
const ctx = canvas.getContext('2d')
canvas.width = window.innerWidth
canvas.height = window.innerHeight

const ripples = []

window.addEventListener('mousemove', (e) => {
  ripples.push({ x: e.clientX, y: e.clientY, r: 0, a: 1 })
})

function render() {
  ctx.clearRect(0, 0, canvas.width, canvas.height)
  for (let i = ripples.length - 1; i >= 0; i--) {
    const rp = ripples[i]
    rp.r += 3
    rp.a -= 0.025

    ctx.beginPath()
    ctx.arc(rp.x, rp.y, rp.r, 0, Math.PI * 2)
    ctx.strokeStyle = `rgba(79, 70, 229, ${rp.a})`
    ctx.lineWidth = 1.5
    ctx.stroke()

    if (rp.a <= 0) ripples.splice(i, 1)
  }
  requestAnimationFrame(render)
}
render()

window.addEventListener('resize', () => {
  canvas.width  = window.innerWidth
  canvas.height = window.innerHeight
})

Image Displacement Shader (Three.js + GLSL)

Classic Codrops image transition: mouse-driven displacement map distorts image on hover. Used extensively in portfolio sites.

import * as THREE from 'three'

// Full-screen image plane with displacement on hover
const scene    = new THREE.Scene()
const camera   = new THREE.OrthographicCamera(-1, 1, 1, -1, 0.1, 10)
camera.position.z = 1
const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true })
renderer.setSize(innerWidth, innerHeight)
document.body.appendChild(renderer.domElement)

const loader = new THREE.TextureLoader()
const image  = loader.load('/hero.jpg')
const disp   = loader.load('/displacement.jpg') // noise/displacement map
disp.wrapS = disp.wrapT = THREE.RepeatWrapping

const material = new THREE.ShaderMaterial({
  uniforms: {
    uTexture:     { value: image },
    uDisplacement: { value: disp },
    uMouse:       { value: new THREE.Vector2(0.5, 0.5) },
    uHover:       { value: 0.0 },
    uScale:       { value: 0.1 },
  },
  vertexShader: `
    varying vec2 vUv;
    void main() { vUv = uv; gl_Position = vec4(position, 1.0); }
  `,
  fragmentShader: `
    uniform sampler2D uTexture;
    uniform sampler2D uDisplacement;
    uniform vec2  uMouse;
    uniform float uHover;
    uniform float uScale;
    varying vec2  vUv;

    void main() {
      vec4 dispMap = texture2D(uDisplacement, vUv);
      float dist = distance(vUv, uMouse);
      float ripple = smoothstep(0.3, 0.0, dist) * uHover;

      vec2 distorted = vUv + dispMap.rg * uScale * ripple;
      gl_FragColor = texture2D(uTexture, distorted);
    }
  `,
})

const mesh = new THREE.Mesh(new THREE.PlaneGeometry(2, 2), material)
scene.add(mesh)

// Mouse tracking
window.addEventListener('mousemove', (e) => {
  material.uniforms.uMouse.value.set(
    e.clientX / innerWidth,
    1 - e.clientY / innerHeight
  )
})

// Hover in/out
const el = document.querySelector('.hero-image')
el.addEventListener('mouseenter', () => gsap.to(material.uniforms.uHover, { value: 1, duration: 0.6 }))
el.addEventListener('mouseleave', () => gsap.to(material.uniforms.uHover, { value: 0, duration: 0.8 }))

;(function animate() { requestAnimationFrame(animate); renderer.render(scene, camera) })()

Noise-Based Blob Shader

import * as THREE from 'three'

// Animated noise blob — Codrops/Bruno Simon style
const geo = new THREE.SphereGeometry(2, 128, 128)
const mat = new THREE.ShaderMaterial({
  uniforms: {
    uTime:      { value: 0 },
    uFrequency: { value: 1.5 },
    uAmplitude: { value: 0.4 },
    uColorA:    { value: new THREE.Color('#4f46e5') },
    uColorB:    { value: new THREE.Color('#e11d48') },
  },
  vertexShader: `
    uniform float uTime;
    uniform float uFrequency;
    uniform float uAmplitude;
    varying float vNoise;

    // Simple 3D noise function
    vec3 mod289(vec3 x) { return x - floor(x*(1./289.))*289.; }
    vec4 mod289(vec4 x) { return x - floor(x*(1./289.))*289.; }
    vec4 permute(vec4 x) { return mod289((x*34.+1.)*x); }
    vec4 taylorInvSqrt(vec4 r) { return 1.79284291400159-.85373472095314*r; }
    float snoise(vec3 v) {
      const vec2 C = vec2(1./6.,1./3.);
      vec4 i = floor(v+dot(v,C.yyy));
      vec3 x0 = v-i+dot(i,C.xxx);
      vec3 g = step(x0.yzx,x0.xyz);
      vec3 l = 1.-g;
      vec3 i1 = min(g.xyz,l.zxy);
      vec3 i2 = max(g.xyz,l.zxy);
      vec3 x1=x0-i1+C.xxx, x2=x0-i2+C.yyy, x3=x0-.5;
      i = mod289(i);
      vec4 p = permute(permute(permute(i.z+vec4(0.,i1.z,i2.z,1.))
        +i.y+vec4(0.,i1.y,i2.y,1.))+i.x+vec4(0.,i1.x,i2.x,1.));
      vec3 ns = .142857142857*C.yyy-.333333333333;
      vec4 j = p-49.*floor(p*ns.z*ns.z);
      vec4 x_ = floor(j*ns.z), y_ = floor(j-7.*x_);
      vec4 x=x_*ns.x+ns.yyyy, y=y_*ns.x+ns.yyyy;
      vec4 h = 1.-abs(x)-abs(y);
      vec4 b0=vec4(x.xy,y.xy), b1=vec4(x.zw,y.zw);
      vec4 s0=floor(b0)*2.+1., s1=floor(b1)*2.+1.;
      vec4 sh = -step(h,vec4(0.));
      vec4 a0=b0.xzyw+s0.xzyw*sh.xxyy, a1=b1.xzyw+s1.xzyw*sh.zzww;
      vec3 p0=vec3(a0.xy,h.x), p1=vec3(a0.zw,h.y),
           p2=vec3(a1.xy,h.z), p3=vec3(a1.zw,h.w);
      vec4 norm = taylorInvSqrt(vec4(dot(p0,p0),dot(p1,p1),dot(p2,p2),dot(p3,p3)));
      p0*=norm.x; p1*=norm.y; p2*=norm.z; p3*=norm.w;
      vec4 m=max(.6-vec4(dot(x0,x0),dot(x1,x1),dot(x2,x2),dot(x3,x3)),0.);
      m=m*m; return 42.*dot(m*m,vec4(dot(p0,x0),dot(p1,x1),dot(p2,x2),dot(p3,x3)));
    }

    void main() {
      vec3 pos = position;
      float n = snoise(pos * uFrequency + uTime * 0.3);
      pos += normal * n * uAmplitude;
      vNoise = n;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);
    }
  `,
  fragmentShader: `
    uniform vec3  uColorA;
    uniform vec3  uColorB;
    varying float vNoise;
    void main() {
      float t = (vNoise + 1.0) * 0.5;
      gl_FragColor = vec4(mix(uColorA, uColorB, t), 1.0);
    }
  `,
})

const blob = new THREE.Mesh(geo, mat)
scene.add(blob)

// In animate: mat.uniforms.uTime.value = clock.getElapsedTime()

Lenis Smooth Scroll + GSAP Integration

Codrops almost always uses Lenis for buttery smooth scrolling. Industry standard pattern as of 2024+.

npm install @studio-freight/lenis
import Lenis from '@studio-freight/lenis'
import gsap from 'gsap'
import ScrollTrigger from 'gsap/ScrollTrigger'

gsap.registerPlugin(ScrollTrigger)

// Init Lenis
const lenis = new Lenis({
  duration: 1.2,
  easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
  smoothWheel: true,
  wheelMultiplier: 1,
  touchMultiplier: 2,
})

// Hook Lenis into GSAP's ticker
gsap.ticker.add((time) => {
  lenis.raf(time * 1000)
})
gsap.ticker.lagSmoothing(0)

// Connect ScrollTrigger
lenis.on('scroll', ScrollTrigger.update)

Scroll Velocity Skew Effect

Elements skew based on scroll velocity. Codrops staple — creates fluid, organic scroll feel.

// Scroll velocity skew — Codrops signature technique
import gsap from 'gsap'
import ScrollTrigger from 'gsap/ScrollTrigger'

const skewElements = document.querySelectorAll('[data-skew]')
const proxy = { skew: 0 }
const clamp = gsap.utils.clamp(-20, 20)

ScrollTrigger.create({
  onUpdate: (self) => {
    const skew = clamp(self.getVelocity() / -300)
    if (Math.abs(skew) > Math.abs(proxy.skew)) {
      proxy.skew = skew
      gsap.to(proxy, {
        skew: 0,
        duration: 0.8,
        ease: 'power3',
        overwrite: true,
        onUpdate: () => {
          skewElements.forEach(el => {
            el.style.transform = `skewY(${proxy.skew}deg)`
          })
        }
      })
    }
  }
})

Parallax Sections

// Codrops parallax — GSAP ScrollTrigger
import gsap from 'gsap'
import ScrollTrigger from 'gsap/ScrollTrigger'

// Parallax images (move slower than scroll)
document.querySelectorAll('[data-parallax-img]').forEach(img => {
  const speed = parseFloat(img.dataset.parallaxImg) || 0.3
  gsap.fromTo(img,
    { yPercent: -20 * speed },
    {
      yPercent: 20 * speed,
      ease: 'none',
      scrollTrigger: {
        trigger: img.closest('section'),
        start: 'top bottom',
        end: 'bottom top',
        scrub: true,
      }
    }
  )
})

// Parallax text layers
gsap.utils.toArray('.parallax-text').forEach((el, i) => {
  const depth = i % 2 === 0 ? -50 : 50
  gsap.fromTo(el,
    { y: depth },
    {
      y: -depth,
      ease: 'none',
      scrollTrigger: {
        trigger: el.closest('section'),
        start: 'top bottom',
        end: 'bottom top',
        scrub: 1,
      }
    }
  )
})

Magnetic Button

// Magnetic button — Codrops signature hover effect
document.querySelectorAll('[data-magnetic]').forEach(btn => {
  const strength = parseFloat(btn.dataset.magnetic) || 0.3

  btn.addEventListener('mousemove', (e) => {
    const rect = btn.getBoundingClientRect()
    const cx = rect.left + rect.width  / 2
    const cy = rect.top  + rect.height / 2
    const dx = (e.clientX - cx) * strength
    const dy = (e.clientY - cy) * strength

    gsap.to(btn, {
      x: dx, y: dy,
      duration: 0.3,
      ease: 'power2.out',
    })

    // Optional: move inner text slightly less
    const inner = btn.querySelector('[data-magnetic-inner]')
    if (inner) gsap.to(inner, { x: dx * 0.4, y: dy * 0.4, duration: 0.3 })
  })

  btn.addEventListener('mouseleave', () => {
    gsap.to(btn, { x: 0, y: 0, duration: 0.6, ease: 'elastic.out(1, 0.4)' })
    const inner = btn.querySelector('[data-magnetic-inner]')
    if (inner) gsap.to(inner, { x: 0, y: 0, duration: 0.6, ease: 'elastic.out(1, 0.4)' })
  })
})
