css — font import + usage
/* Google Fonts import */
@import url('https://fonts.googleapis.com/css2?family=Roboto:wght@100;300;400;500;700;900&family=Caveat:wght@500;600&display=swap');

/* Usage pattern */
body {
  font-family: 'Roboto', sans-serif;
  font-weight: 400;
}

h1, h2 {
  font-family: 'Roboto', sans-serif;
  font-weight: 700;
}

/* Accent / handwritten callouts */
.accent-text, .tagline {
  font-family: 'Caveat', cursive;
  font-weight: 600;
}

/* Roboto weight scale used */
/* 100 — thin decorative */
/* 300 — light body text */
/* 400 — regular */
/* 500 — medium UI labels */
/* 700 — bold headings */
/* 900 — black hero headings */

/* Caveat weight scale used */
/* 500 — medium cursive */
/* 600 — semibold cursive (primary use) */

import { Caveat, Roboto } from 'next/font/google';

const caveat = Caveat({ subsets: ['latin'], weight: ['500', '600'] });
const roboto = Roboto({ subsets: ['latin'], weight: ['400', '700', '900'] });

export function HeroSection() {
  return (
    <section className="relative min-h-screen flex items-center bg-navy-dark">
      {/* Background kitchen image with overlay */}
      <div className="absolute inset-0 bg-black/50 z-0" />
      <div className="relative z-10 container mx-auto px-6">
        {/* Caveat accent line */}
        <p className={`${caveat.className} text-2xl text-gold-400 mb-4`}>
          Custom Renovations
        </p>
        {/* Roboto black headline */}
        <h1 className={`${roboto.className} font-black text-5xl md:text-7xl text-white leading-tight mb-6`}>
          Build Your<br/>
          <span className="text-gold-400">Dream Kitchen</span><br/>
          Today
        </h1>
        {/* Credibility sub */}
        <p className="text-lg text-white/80 mb-8 max-w-xl">
          5-star rated Canadian kitchen renovations that fit your
          lifestyle and budget.
        </p>
        <div className="flex gap-4">
          <Button size="lg" className="bg-gold-500 hover:bg-gold-400 text-black">
            Get Free Quote
          </Button>
          <Button size="lg" variant="outline" className="text-white border-white">
            See Our Work
          </Button>
        </div>
      </div>
    </section>
  );
}
