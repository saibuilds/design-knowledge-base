How Webflow Interactions Work

Webflow interactions output clean CSS animations + vanilla JS (no jQuery). Understanding the patterns lets you recreate them in any framework.

Interaction Types

Page Load triggerRuns once on DOMContentLoaded. Used for entrance animations: fade-in, slide-up, staggered reveals.
Scroll Into ViewIntersectionObserver-based. Triggers when element enters viewport. Stagger delay on child elements.
Scroll PositionMaps scroll percentage to element properties (opacity, transform, color). Equivalent to GSAP ScrollTrigger scrub.
Mouse Move (global)Maps cursor X/Y to element rotation/translation. Creates parallax and magnetic effects.
Hover triggerMouse enter/leave. Typically animates scale, color, border, clip-path on hover.
Click triggerToggle states. Used for accordions, modals, nav menus, tab switches.
While ScrollingSticky elements, progress bars, parallax layers. Tied to scroll progress 0-100%.

Exported Interaction Code Pattern

Webflow exports interactions as inline attribute-driven JS. Here's the reconstructed pattern:

// Webflow scroll-into-view pattern (reconstructed)
// data-w-id attributes map to animation timelines

const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      entry.target.classList.add('visible')
      // Webflow applies inline style transforms:
      // opacity: 0 -> 1, transform: translateY(20px) -> translateY(0)
    }
  })
}, { threshold: 0.15 })

document.querySelectorAll('[data-animate]').forEach(el => observer.observe(el))

/* Webflow stagger entrance — recreate in plain CSS */
[data-animate] {
  opacity: 0;
  transform: translateY(24px);
  transition: opacity 0.6s ease, transform 0.6s ease;
}
[data-animate].visible { opacity: 1; transform: translateY(0); }

/* Stagger children */
[data-animate].visible > *:nth-child(1) { transition-delay: 0.0s; }
[data-animate].visible > *:nth-child(2) { transition-delay: 0.1s; }
[data-animate].visible > *:nth-child(3) { transition-delay: 0.2s; }
[data-animate].visible > *:nth-child(4) { transition-delay: 0.3s; }
[data-animate].visible > *:nth-child(5) { transition-delay: 0.4s; }

/* GSAP handles scale + translateY per card on scroll */

// Stacked cards GSAP implementation
import gsap from 'gsap'
import ScrollTrigger from 'gsap/ScrollTrigger'
gsap.registerPlugin(ScrollTrigger)

const cards = document.querySelectorAll('.card')

cards.forEach((card, i) => {
  if (i === cards.length - 1) return // last card stays

  gsap.to(card, {
    scale: 0.9 - (i * 0.05),
    yPercent: -10 * i,
    scrollTrigger: {
      trigger: '.cards-container',
      start: `${(i / cards.length) * 100}% top`,
      end: `${((i + 1) / cards.length) * 100}% top`,
      scrub: 1,
    }
  })
})

Horizontal Scroll Section

// Webflow-style horizontal scroll (GSAP)
import gsap from 'gsap'
import ScrollTrigger from 'gsap/ScrollTrigger'
gsap.registerPlugin(ScrollTrigger)

const wrapper = document.querySelector('.h-scroll-wrapper')
const track   = document.querySelector('.h-scroll-track')

const totalWidth = track.scrollWidth - window.innerWidth

gsap.to(track, {
  x: -totalWidth,
  ease: 'none',
  scrollTrigger: {
    trigger: wrapper,
    pin: true,
    scrub: 1,
    start: 'top top',
    end: () => `+=${totalWidth}`,
    invalidateOnRefresh: true,
  }
})

Mouse Parallax (Global)

// Webflow mouse-move parallax — reconstructed pattern
document.addEventListener('mousemove', (e) => {
  const cx = window.innerWidth  / 2
  const cy = window.innerHeight / 2
  const dx = (e.clientX - cx) / cx  // -1 to 1
  const dy = (e.clientY - cy) / cy  // -1 to 1

  // Parallax layers with different depths
  document.querySelectorAll('[data-parallax]').forEach(el => {
    const depth = parseFloat(el.dataset.parallax) || 1
    const tx = dx * depth * 20
    const ty = dy * depth * 20
    el.style.transform = `translate(${tx}px, ${ty}px)`
  })

  // Tilt card effect
  document.querySelectorAll('[data-tilt]').forEach(el => {
    const rect = el.getBoundingClientRect()
    const lx = (e.clientX - rect.left) / rect.width  - 0.5  // -0.5 to 0.5
    const ly = (e.clientY - rect.top)  / rect.height - 0.5
    el.style.transform = `perspective(800px) rotateY(${lx * 15}deg) rotateX(${-ly * 15}deg)`
  })
})

Scroll-Triggered Fade + Slide

// Webflow scroll-into-view — vanilla IntersectionObserver
const io = new IntersectionObserver((entries) => {
  entries.forEach(({ target, isIntersecting }) => {
    if (!isIntersecting) return
    target.style.opacity = '1'
    target.style.transform = 'translateY(0) scale(1)'
    io.unobserve(target) // fire once
  })
}, { threshold: 0.1, rootMargin: '0px 0px -60px 0px' })

document.querySelectorAll('[data-reveal]').forEach((el, i) => {
  el.style.opacity = '0'
  el.style.transform = 'translateY(32px) scale(0.97)'
  el.style.transition = `opacity 0.7s ease ${i * 0.08}s, transform 0.7s ease ${i * 0.08}s`
  io.observe(el)
})
