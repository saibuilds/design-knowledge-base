// app/garden-suite/[neighbourhood]/page.tsx — dynamic neighbourhood pages
import type { Metadata } from 'next'

type Props = { params: { neighbourhood: string } }

const neighbourhoodMap: Record<string, string> = {
  'leslieville':   'Leslieville',
  'east-york':     'East York',
  'the-beaches':   'The Beaches',
  'danforth':      'Danforth',
  'north-york':    'North York',
  'scarborough':   'Scarborough',
  'etobicoke':     'Etobicoke',
}

export async function generateStaticParams() {
  return Object.keys(neighbourhoodMap).map((n) => ({ neighbourhood: n }))
}

export async function generateMetadata({ params }: Props): Promise<Metadata> {
  const name = neighbourhoodMap[params.neighbourhood] ?? params.neighbourhood
  return {
    title: `Garden Suite Builder in ${name}, Toronto | YourCompany`,
    description: `Build a permit-approved garden suite in ${name}, Toronto. Fixed pricing, in-house permit handling, free feasibility assessment. See if your property is eligible.`,
    alternates: {
      canonical: `https://yoursite.ca/garden-suite/${params.neighbourhood}`,
    },
    openGraph: {
      title: `Garden Suite Builder in ${name} | YourCompany`,
      description: `Permit-approved garden suite construction in ${name}, Toronto. Free consultation.`,
      url: `https://yoursite.ca/garden-suite/${params.neighbourhood}`,
    },
  }
}

## 8. Competitor Content Gaps to Exploit

**Gap 1: Real cost breakdowns with line items**
- Most sites say "contact us for a quote" — no actual numbers
- Publish a detailed cost breakdown table (foundation, framing, plumbing, electrical, permits, design, landscaping)
- Targets the "garden suite cost Toronto" keyword cluster directly

**Gap 2: Neighbourhood-specific eligibility pages**
- "Can I build a garden suite in [East York / Scarborough / North York]?"
- Each neighbourhood has different lot sizes and zoning nuances
- Low competition, high conversion intent — people searching this are ready to act

**Gap 3: Financing options consolidated in one page**
- CMHC Secondary Suite Loan (up to $40,000 at subsidized rate)
- CMHC MLI Select program
- HELOC strategy
- Toronto's Housing Now initiative
- No competitor covers all options in one place

**Gap 4: Failure / risk content**
- "What Happens If You Build a Garden Suite Without a Permit in Toronto?"
- "5 Garden Suite Permit Mistakes Toronto Homeowners Make"
- Builds trust, ranks for worry-driven long-tail queries

**Gap 5: End-to-end project case studies**
- Document a project from consultation → design → permit approval date → construction start/end → first rent received
- Most competitors show finished photos only — full case studies build trust and dwell time
