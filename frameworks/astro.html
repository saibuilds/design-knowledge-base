Astro Reference

Islands architecture, content collections, and static site patterns for Astro 4/5

When to Use Astro vs Next.js

Use Astro for
Marketing sites, blogs, portfolios, docs — mostly static content
Use Next.js for
Apps with auth, user data, frequent server state, full-stack APIs
Astro strength
Zero JS by default — ships HTML/CSS only unless you opt in
Astro islands
Hydrate only the components that need interactivity

Bootstrap

Create + configure Astro project
npm create astro@latest my-site
# Choose: Empty / With blog / Use TypeScript → Strict

cd my-site

# Add React islands (for interactive components)
npx astro add react

# Add Tailwind
npx astro add tailwind

# Add MDX (for content with components)
npx astro add mdx

# Add sitemap (automatic)
npx astro add sitemap

Islands Architecture

React component with client directive
---
// In an .astro file
import Counter from '../components/Counter.tsx';
import HeavyChart from '../components/HeavyChart.tsx';
---

<!-- client:load — hydrate immediately on page load -->
<Counter client:load />

<!-- client:idle — hydrate when browser is idle -->
<Counter client:idle />

<!-- client:visible — hydrate when scrolled into view -->
<HeavyChart client:visible />

<!-- client:only="react" — skip SSR entirely, render client-side only -->
<ThreeCanvas client:only="react" />

<!-- No directive = 0 JavaScript shipped -->
<StaticCard />
Use client:only="react" for Three.js / R3F components — they need window and WebGL which aren't available at build time.

Content Collections

Define schema + query posts
// src/content/config.ts
import { defineCollection, z } from 'astro:content';

const blog = defineCollection({
  type: 'content',
  schema: z.object({
    title: z.string(),
    date: z.date(),
    description: z.string(),
    image: z.string().optional(),
    tags: z.array(z.string()).default([])
  })
});

export const collections = { blog };

// src/pages/blog/index.astro
---
import { getCollection } from 'astro:content';

const posts = await getCollection('blog');
const sorted = posts.sort((a, b) =>
  b.data.date.valueOf() - a.data.date.valueOf()
);
---

{sorted.map(post => (
  <article>
    <a href={`/blog/${post.slug}`}>{post.data.title}</a>
    <p>{post.data.description}</p>
  </article>
))}
Dynamic route for blog posts
---
// src/pages/blog/[slug].astro
import { getCollection, getEntry } from 'astro:content';

export async function getStaticPaths() {
  const posts = await getCollection('blog');
  return posts.map(post => ({
    params: { slug: post.slug },
    props: { post }
  }));
}

const { post } = Astro.props;
const { Content } = await post.render();
---

<article>
  <h1>{post.data.title}</h1>
  <Content />
</article>

Metadata & SEO

BaseLayout with SEO props
---
// src/layouts/BaseLayout.astro
interface Props {
  title: string;
  description: string;
  image?: string;
  canonical?: string;
}

const {
  title,
  description,
  image = '/og-default.jpg',
  canonical = Astro.url.href
} = Astro.props;

---

<head>
  <meta charset="UTF-8" />
  <meta name="viewport" content="width=device-width" />
  <title>{title}</title>
  <meta name="description" content={description} />
  <link rel="canonical" href={canonical} />

  <!-- Open Graph -->
  <meta property="og:title" content={title} />
  <meta property="og:description" content={description} />
  <meta property="og:image" content={`${siteUrl}${image}`} />
  <meta property="og:type" content="website" />

  <!-- Twitter -->
  <meta name="twitter:card" content="summary_large_image" />
</head>
<body>
  <slot />
</body>
</html>

Built-in page transitions

---

<html>

  <ViewTransitions /> <!-- enables SPA-like transitions -->
</head>

  <slot />
</body>
</html>

<!-- Animate specific elements with transition:name -->
<img
  src={post.data.image}
  transition:name={`hero-${post.slug}`}
  transition:animate="slide"
/>

<!-- Custom transition animations (CSS) -->
<style>
  @keyframes fade-in {
    from { opacity: 0; transform: translateY(10px); }
  }
  @keyframes fade-out {
    to { opacity: 0; transform: translateY(-10px); }
  }
  ::view-transition-old(root) { animation: fade-out 0.2s ease; }
  ::view-transition-new(root) { animation: fade-in 0.2s ease; }
</style>
Astro View Transitions work across the full MPA — no React needed. They're based on the browser's native View Transitions API with a polyfill for older browsers.

Server Endpoints (API Routes)

Contact form API endpoint
// src/pages/api/contact.ts
import type { APIRoute } from 'astro';

export const POST: APIRoute = async ({ request }) => {
  const data = await request.formData();
  const name = data.get('name');
  const email = data.get('email');
  const message = data.get('message');

  if (!name || !email || !message) {
    return new Response(JSON.stringify({ error: 'Missing fields' }), {
      status: 400,
      headers: { 'Content-Type': 'application/json' }
    });
  }

  // Send email, save to DB, etc.
  // await sendEmail({ name, email, message });

  return new Response(JSON.stringify({ success: true }), {
    status: 200,
    headers: { 'Content-Type': 'application/json' }
  });
};

// astro.config.mjs — must enable SSR for server endpoints
// output: 'hybrid' (static by default, server where needed)
// or output: 'server' (all server-rendered)

Astro Config

astro.config.mjs — common setup
// astro.config.mjs
import { defineConfig } from 'astro/config';
import react from '@astrojs/react';
import tailwind from '@astrojs/tailwind';
import sitemap from '@astrojs/sitemap';
import mdx from '@astrojs/mdx';

export default defineConfig({
  site: 'https://mysite.com',   // required for sitemap + canonical URLs
  output: 'hybrid',             // static by default, opt-in to SSR per page
  integrations: [
    react(),
    tailwind({ applyBaseStyles: false }),
    sitemap(),
    mdx()
  ],
  image: {
    // Allow external image domains
    domains: ['images.unsplash.com']
  }
});
