Lenis Smooth Scroll

Lenis is a lightweight smooth-scroll library by Darkroom Engineering. It provides buttery inertia-based scroll that feels native — not the janky CSS scroll-behavior: smooth. Integrates seamlessly with GSAP ScrollTrigger.

Basic Setup

Vanilla JS setup — create a Lenis instance and feed it to requestAnimationFrame.

javascript — vanilla
import Lenis from 'lenis';

const lenis = new Lenis({
  duration: 1.2,
  easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
  smoothWheel: true,
  wheelMultiplier: 1,
  touchMultiplier: 2,
  infinite: false,
});

// RAF loop — Lenis needs to be called every frame
function raf(time) {
  lenis.raf(time);
  requestAnimationFrame(raf);
}
requestAnimationFrame(raf);

// Stop/start
lenis.stop();
lenis.start();

// Destroy (remove all event listeners)
lenis.destroy();

Methods & Events

javascript
// Scroll to target
lenis.scrollTo('#target');                // CSS selector
lenis.scrollTo(element);                  // DOM element
lenis.scrollTo(500);                      // pixel value
lenis.scrollTo('top');                    // 'top' | 'bottom' | 'left' | 'right'
lenis.scrollTo(500, {
  offset: -80,        // offset from target (e.g. fixed header height)
  duration: 1.5,
  easing: t => t,
  immediate: false,   // skip animation
  lock: true,         // lock scroll during animation
  onComplete: () => console.log('arrived')
});

// Events
lenis.on('scroll', ({ scroll, limit, velocity, direction, progress }) => {
  console.log(scroll);     // current scroll position
  console.log(limit);      // max scroll position
  console.log(velocity);   // scroll speed
  console.log(direction);  // 1 = down, -1 = up
  console.log(progress);   // 0-1 progress through page
});
lenis.off('scroll', handler);  // remove listener

// Properties
lenis.scroll;       // current scroll value
lenis.limit;        // max scroll value
lenis.velocity;     // scroll velocity
lenis.direction;    // 1 | -1
lenis.progress;     // 0-1
lenis.isScrolling;  // boolean
lenis.isStopped;    // boolean

GSAP ScrollTrigger Integration

The most important integration — without this, ScrollTrigger won't respond correctly to Lenis scroll events.

javascript — vanilla GSAP + Lenis
import Lenis from 'lenis';
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
gsap.registerPlugin(ScrollTrigger);

const lenis = new Lenis({
  duration: 1.2,
  easing: t => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
});

// Method 1: Feed Lenis into GSAP ticker (recommended)
gsap.ticker.add((time) => {
  lenis.raf(time * 1000);  // GSAP time is in seconds, Lenis needs ms
});
gsap.ticker.lagSmoothing(0);  // prevent GSAP from skipping frames

// Tell ScrollTrigger about Lenis scroll updates
lenis.on('scroll', ScrollTrigger.update);

// IMPORTANT: disable ScrollTrigger's normalizeScroll to avoid conflicts
// ScrollTrigger.normalizeScroll(false); // usually not needed with this approach

// Refresh ScrollTrigger after everything loads
window.addEventListener('load', () => ScrollTrigger.refresh());
Common issue: ScrollTrigger positions are calculated based on the native scroll position, but Lenis uses a virtual scroll. The lenis.on('scroll', ScrollTrigger.update) call keeps them in sync. Without it, triggers fire at wrong positions.

React / Next.js Setup (Manual)

jsx — Next.js App Router provider
'use client';
// app/providers/smooth-scroll.jsx
import { createContext, useContext, useEffect, useRef } from 'react';
import Lenis from 'lenis';
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
gsap.registerPlugin(ScrollTrigger);

const LenisContext = createContext(null);
export const useLenis = () => useContext(LenisContext);

export function SmoothScrollProvider({ children }) {
  const lenisRef = useRef(null);

  useEffect(() => {
    const lenis = new Lenis({
      duration: 1.2,
      easing: t => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
      smoothWheel: true,
    });
    lenisRef.current = lenis;

    // GSAP ticker integration
    function onRaf(time) { lenis.raf(time * 1000); }
    gsap.ticker.add(onRaf);
    gsap.ticker.lagSmoothing(0);
    lenis.on('scroll', ScrollTrigger.update);

    return () => {
      gsap.ticker.remove(onRaf);
      lenis.destroy();
    };
  }, []);

  return (
    <LenisContext.Provider value={lenisRef}>
      {children}
    </LenisContext.Provider>
  );
}

// Usage in app/layout.jsx:
// <SmoothScrollProvider>{children}</SmoothScrollProvider>

// Usage in a component:
// const lenisRef = useLenis();
// lenisRef.current?.scrollTo('#section', { offset: -80 });

react-lenis Package

The official React wrapper — simplest way to add Lenis in a React/Next.js project.

bash
npm install lenis  # react-lenis is bundled in the lenis package
jsx — react-lenis usage
import { ReactLenis, useLenis } from 'lenis/react';

// Wrap your app (or layout)
export default function Layout({ children }) {
  return (
    <ReactLenis
      root
      options={{
        duration: 1.2,
        easing: t => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
        smoothWheel: true,
      }}
    >
      {children}
    </ReactLenis>
  );
}

// Access lenis instance anywhere
function ScrollToButton() {
  const lenis = useLenis(({ scroll }) => {
    // This callback fires on every scroll
    // console.log(scroll);
  });

  return (
    <button onClick={() => lenis?.scrollTo('#contact', { offset: -80 })}>
      Scroll to Contact
    </button>
  );
}

// Connect to GSAP ScrollTrigger when using react-lenis
import { useEffect } from 'react';
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';

function GSAPScrollTriggerSync() {
  const lenis = useLenis();

  useEffect(() => {
    if (!lenis) return;
    const rafFn = (time) => lenis.raf(time * 1000);
    gsap.ticker.add(rafFn);
    gsap.ticker.lagSmoothing(0);
    lenis.on('scroll', ScrollTrigger.update);
    return () => {
      gsap.ticker.remove(rafFn);
      lenis.off('scroll', ScrollTrigger.update);
    };
  }, [lenis]);

  return null;
}

Tips & Gotchas

iFrame / embeds: Use the prevent option to prevent Lenis from intercepting scroll in embedded content:
prevent: node => node.nodeName === 'IFRAME'

Fixed elements: Elements with position: fixed work normally — Lenis only affects the scroll container, not fixed positioning.

Anchor links: Native anchor links (#id) still work, but scroll is animated by Lenis. For instant jump, use lenis.scrollTo('#id', { immediate: true }).

Overflow hidden containers: If a parent has overflow:hidden, Lenis won't scroll inside it by default. Create a separate Lenis instance for that container, or use the wrapper/content options.

Mobile: Lenis uses native scroll behavior on mobile by default for best performance. Only touch multiplier is applied.

Next.js SSR: Always check typeof window !== 'undefined' or use useEffect — Lenis requires browser APIs.
javascript — stop scroll during modal/overlay
// Lock scroll when modal opens
function openModal() {
  lenis.stop();
  document.body.style.overflow = 'hidden'; // belt and suspenders
}

function closeModal() {
  lenis.start();
  document.body.style.overflow = '';
}

// Scroll to section with nav offset
function scrollToSection(id) {
  const navHeight = document.querySelector('nav').offsetHeight;
  lenis.scrollTo(id, {
    offset: -navHeight - 16,  // account for nav + extra gap
    duration: 1.2,
    easing: t => Math.min(1, 1.001 - Math.pow(2, -10 * t))
  });
}
