1. GSAP SplitText Character Reveal

Splits an h1 into individual characters using GSAP SplitText, then staggers them in on scroll via ScrollTrigger.

GSAP Club SplitText plugin ScrollTrigger plugin React
Requires GSAP Club membership for SplitText. Install: gsap + register both plugins.
// SplitTextReveal.tsx
import { useEffect, useRef } from "react";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";
import { SplitText } from "gsap/SplitText";

gsap.registerPlugin(ScrollTrigger, SplitText);

interface SplitTextRevealProps {
  text: string;
  className?: string;
}

export function SplitTextReveal({ text, className = "" }: SplitTextRevealProps) {
  const headingRef = useRef<HTMLHeadingElement>(null);

  useEffect(() => {
    const el = headingRef.current;
    if (!el) return;

    // Split into characters — SplitText wraps each char in a span
    const split = new SplitText(el, { type: "chars,words" });
    const chars = split.chars;

    // Set overflow hidden on the parent so chars don't flash before animating
    gsap.set(el, { overflow: "hidden" });

    const tl = gsap.fromTo(
      chars,
      {
        y: "110%",
        opacity: 0,
        rotateZ: 8,
      },
      {
        y: "0%",
        opacity: 1,
        rotateZ: 0,
        duration: 0.7,
        ease: "power3.out",
        stagger: 0.025,
        scrollTrigger: {
          trigger: el,
          start: "top 85%",
          toggleActions: "play none none reverse",
        },
      }
    );

    return () => {
      tl.kill();
      split.revert(); // restore original HTML
      ScrollTrigger.getAll().forEach((st) => st.kill());
    };
  }, [text]);

  return (
    <h1
      ref={headingRef}
      className={`text-5xl font-bold tracking-tight ${className}`}
    >
      {text}
    </h1>
  );
}

// <SplitTextReveal text="Hello World" className="text-white" />

2. Magnetic Button

Mouse proximity pulls the button toward the cursor (attract). On leave, it springs back. Uses lerp for smooth follow.

Vanilla JS React hook No deps 
// MagneticButton.tsx
import { useRef, useCallback, MouseEvent } from "react";

interface MagneticButtonProps {
  children: React.ReactNode;
  strength?: number; // 0–1, how strongly it pulls (default 0.4)
  className?: string;
}

export function MagneticButton({
  children,
  strength = 0.4,
  className = "",
}: MagneticButtonProps) {
  const btnRef = useRef<HTMLButtonElement>(null);
  const rafRef = useRef<number>(0);
  const pos = useRef({ x: 0, y: 0 });
  const target = useRef({ x: 0, y: 0 });

  const lerp = (a: number, b: number, t: number) => a + (b - a) * t;

  const animate = useCallback(() => {
    pos.current.x = lerp(pos.current.x, target.current.x, 0.1);
    pos.current.y = lerp(pos.current.y, target.current.y, 0.1);

    if (btnRef.current) {
      btnRef.current.style.transform = `translate(${pos.current.x}px, ${pos.current.y}px)`;
    }

    // Stop when close enough
    const dx = Math.abs(pos.current.x - target.current.x);
    const dy = Math.abs(pos.current.y - target.current.y);
    if (dx > 0.1 || dy > 0.1) {
      rafRef.current = requestAnimationFrame(animate);
    }
  }, []);

  const handleMouseMove = (e: MouseEvent<HTMLButtonElement>) => {
    const rect = btnRef.current!.getBoundingClientRect();
    const centerX = rect.left + rect.width / 2;
    const centerY = rect.top + rect.height / 2;
    const deltaX = (e.clientX - centerX) * strength;
    const deltaY = (e.clientY - centerY) * strength;

    target.current = { x: deltaX, y: deltaY };

    cancelAnimationFrame(rafRef.current);
    rafRef.current = requestAnimationFrame(animate);
  };

  const handleMouseLeave = () => {
    target.current = { x: 0, y: 0 };
    cancelAnimationFrame(rafRef.current);
    rafRef.current = requestAnimationFrame(animate);
  };

  return (
    <button
      ref={btnRef}
      onMouseMove={handleMouseMove}
      onMouseLeave={handleMouseLeave}
      className={`relative inline-flex items-center justify-center px-8 py-4 rounded-full
        bg-white text-black font-semibold text-sm transition-shadow duration-300
        hover:shadow-[0_0_40px_rgba(255,255,255,0.2)] ${className}`}
      style={{ willChange: "transform" }}
    >
      {children}
    </button>
  );
}

// Usage:
// <MagneticButton strength={0.5}>Click me</MagneticButton>

3. Custom Cursor

Smooth lerp cursor dot + ring. Scales on hover over interactive elements. Changes color when over dark/light sections via data attribute.

React context rAF lerp data-cursor attribute 
// CustomCursor.tsx
import { useEffect, useRef, useState } from "react";

export function CustomCursor() {
  const dotRef = useRef<HTMLDivElement>(null);
  const ringRef = useRef<HTMLDivElement>(null);

  const mouse = useRef({ x: -100, y: -100 });
  const dotPos = useRef({ x: -100, y: -100 });
  const ringPos = useRef({ x: -100, y: -100 });
  const rafRef = useRef<number>(0);

  const [isHovering, setIsHovering] = useState(false);
  const [cursorColor, setCursorColor] = useState("white");

  const lerp = (a: number, b: number, t: number) => a + (b - a) * t;

  useEffect(() => {
    const onMouseMove = (e: globalThis.MouseEvent) => {
      mouse.current = { x: e.clientX, y: e.clientY };

      // Detect section color
      const el = document.elementFromPoint(e.clientX, e.clientY);
      const section = el?.closest("[data-cursor-color]");
      const color = section?.getAttribute("data-cursor-color") ?? "white";
      setCursorColor(color);
    };

    const onMouseOver = (e: globalThis.MouseEvent) => {
      const el = e.target as HTMLElement;
      if (el.matches("a, button, [data-cursor-hover]")) {
        setIsHovering(true);
      }
    };

    const onMouseOut = (e: globalThis.MouseEvent) => {
      const el = e.target as HTMLElement;
      if (el.matches("a, button, [data-cursor-hover]")) {
        setIsHovering(false);
      }
    };

    window.addEventListener("mousemove", onMouseMove);
    window.addEventListener("mouseover", onMouseOver);
    window.addEventListener("mouseout", onMouseOut);

    const tick = () => {
      // Dot follows faster
      dotPos.current.x = lerp(dotPos.current.x, mouse.current.x, 0.5);
      dotPos.current.y = lerp(dotPos.current.y, mouse.current.y, 0.5);
      // Ring follows slower
      ringPos.current.x = lerp(ringPos.current.x, mouse.current.x, 0.12);
      ringPos.current.y = lerp(ringPos.current.y, mouse.current.y, 0.12);

      if (dotRef.current) {
        dotRef.current.style.transform = `translate(${dotPos.current.x - 4}px, ${dotPos.current.y - 4}px)`;
      }
      if (ringRef.current) {
        ringRef.current.style.transform = `translate(${ringPos.current.x - 20}px, ${ringPos.current.y - 20}px)`;
      }

      rafRef.current = requestAnimationFrame(tick);
    };

    rafRef.current = requestAnimationFrame(tick);

    return () => {
      window.removeEventListener("mousemove", onMouseMove);
      window.removeEventListener("mouseover", onMouseOver);
      window.removeEventListener("mouseout", onMouseOut);
      cancelAnimationFrame(rafRef.current);
    };
  }, []);

  const dotColor = cursorColor === "dark" ? "#000" : "#fff";
  const ringBorder = cursorColor === "dark" ? "rgba(0,0,0,0.6)" : "rgba(255,255,255,0.6)";

  return (
    <>
      {/* Dot */}
      <div
        ref={dotRef}
        style={{
          position: "fixed",
          top: 0,
          left: 0,
          width: 8,
          height: 8,
          borderRadius: "50%",
          background: dotColor,
          pointerEvents: "none",
          zIndex: 9999,
          transition: "background 0.3s, transform 0.05s",
          willChange: "transform",
        }}
      />
      {/* Ring */}
      <div
        ref={ringRef}
        style={{
          position: "fixed",
          top: 0,
          left: 0,
          width: 40,
          height: 40,
          borderRadius: "50%",
          border: `1.5px solid ${ringBorder}`,
          pointerEvents: "none",
          zIndex: 9998,
          transition: "border-color 0.3s, width 0.25s ease, height 0.25s ease",
          willChange: "transform",
          ...(isHovering
            ? { width: 60, height: 60, mixBlendMode: "difference" as const }
            : {}),
        }}
      />
    </>
  );
}

4. Horizontal Scroll Section

GSAP ScrollTrigger pins the section and scrubs a horizontal translateX, creating a sideways scroll experience.

GSAP ScrollTrigger React 
// HorizontalScroll.tsx
import { useEffect, useRef } from "react";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";

gsap.registerPlugin(ScrollTrigger);

interface HorizontalScrollProps {
  children: React.ReactNode;
}

export function HorizontalScroll({ children }: HorizontalScrollProps) {
  const sectionRef = useRef<HTMLElement>(null);
  const trackRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const section = sectionRef.current;
    const track = trackRef.current;
    if (!section || !track) return;

    // Total scroll distance = track width minus viewport width
    const getScrollAmount = () => -(track.scrollWidth - window.innerWidth);

    const ctx = gsap.context(() => {
      gsap.to(track, {
        x: getScrollAmount,
        ease: "none",
        scrollTrigger: {
          trigger: section,
          start: "top top",
          end: () => `+=${track.scrollWidth - window.innerWidth}`,
          pin: true,
          scrub: 1,
          anticipatePin: 1,
          invalidateOnRefresh: true,
        },
      });
    }, section);

    return () => ctx.revert();
  }, []);

  return (
    <section
      ref={sectionRef}
      className="relative overflow-hidden h-screen"
    >
      {/* Track: flex row of panels, each panel = 100vw */}
      <div
        ref={trackRef}
        className="flex h-full will-change-transform"
        style={{ width: "max-content" }}
      >
        {children}
      </div>
    </section>
  );
}

// Panel helper
export function HScrollPanel({
  children,
  bg = "bg-neutral-900",
}: {
  children: React.ReactNode;
  bg?: string;
}) {
  return (
    <div
      className={`w-screen h-full flex items-center justify-center flex-shrink-0 ${bg}`}
    >
      {children}
    </div>
  );
}

5. Page Transitions (Framer Motion)

AnimatePresence wraps pages, each page slides in/out. Works with Next.js App Router or React Router.

framer-motion Next.js / React Router AnimatePresence 
// PageTransition.tsx  (Next.js App Router)
"use client";
import { AnimatePresence, motion } from "framer-motion";
import { usePathname } from "next/navigation";

const variants = {
  initial: { opacity: 0, y: 20, filter: "blur(4px)" },
  enter: {
    opacity: 1,
    y: 0,
    filter: "blur(0px)",
    transition: { duration: 0.45, ease: [0.25, 0.46, 0.45, 0.94] },
  },
  exit: {
    opacity: 0,
    y: -20,
    filter: "blur(4px)",
    transition: { duration: 0.3, ease: [0.25, 0.46, 0.45, 0.94] },
  },
};

export function PageTransition({ children }: { children: React.ReactNode }) {
  const pathname = usePathname();

  return (
    <AnimatePresence mode="wait">
      <motion.div
        key={pathname}
        variants={variants}
        initial="initial"
        animate="enter"
        exit="exit"
        className="min-h-screen"
      >
        {children}
      </motion.div>
    </AnimatePresence>
  );
}

// ----------------------------------------------------------
// Slide variant (left ↔ right based on nav direction)
// ----------------------------------------------------------
// slideVariants.ts
export const slideVariants = (direction: "left" | "right") => ({
  initial: { x: direction === "right" ? "100%" : "-100%", opacity: 0 },
  enter: {
    x: 0,
    opacity: 1,
    transition: { duration: 0.5, ease: [0.76, 0, 0.24, 1] },
  },
  exit: {
    x: direction === "right" ? "-100%" : "100%",
    opacity: 0,
    transition: { duration: 0.4, ease: [0.76, 0, 0.24, 1] },
  },
});

// ----------------------------------------------------------
// Curtain wipe variant
// ----------------------------------------------------------
export function CurtainTransition({ children }: { children: React.ReactNode }) {
  const pathname = usePathname();

  return (
    <AnimatePresence mode="wait">
      <motion.div key={pathname} className="relative min-h-screen">
        {/* Curtain overlay */}
        <motion.div
          className="fixed inset-0 z-50 bg-black pointer-events-none origin-top"
          initial={{ scaleY: 1 }}
          animate={{ scaleY: 0, transition: { duration: 0.6, ease: [0.76, 0, 0.24, 1] } }}
          exit={{ scaleY: 1, transition: { duration: 0.5, ease: [0.76, 0, 0.24, 1] } }}
        />
        {children}
      </motion.div>
    </AnimatePresence>
  );
}

7. Clip-Path Reveal

Elements reveal from bottom to top using clip-path inset animation triggered on scroll via ScrollTrigger or Intersection Observer.

GSAP ScrollTrigger clip-path 
// ClipReveal.tsx
import { useEffect, useRef } from "react";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";

gsap.registerPlugin(ScrollTrigger);

type Direction = "bottom" | "top" | "left" | "right";

interface ClipRevealProps {
  children: React.ReactNode;
  direction?: Direction;
  duration?: number;
  delay?: number;
  className?: string;
}

const clipStart: Record<Direction, string> = {
  bottom: "inset(100% 0% 0% 0%)",
  top:    "inset(0% 0% 100% 0%)",
  left:   "inset(0% 100% 0% 0%)",
  right:  "inset(0% 0% 0% 100%)",
};

export function ClipReveal({
  children,
  direction = "bottom",
  duration = 0.9,
  delay = 0,
  className = "",
}: ClipRevealProps) {
  const ref = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const el = ref.current;
    if (!el) return;

    gsap.fromTo(
      el,
      { clipPath: clipStart[direction] },
      {
        clipPath: "inset(0% 0% 0% 0%)",
        duration,
        delay,
        ease: "power4.inOut",
        scrollTrigger: {
          trigger: el,
          start: "top 88%",
          toggleActions: "play none none reverse",
        },
      }
    );

    return () => {
      ScrollTrigger.getAll().forEach((st) => st.kill());
    };
  }, [direction, duration, delay]);

  return (
    <div
      ref={ref}
      className={className}
      style={{ clipPath: clipStart[direction], willChange: "clip-path" }}
    >
      {children}
    </div>
  );
}

8. Parallax Image

Image moves at a different speed than the scroll using GSAP scrub, creating a depth effect. Image is slightly oversized to cover gaps.

GSAP ScrollTrigger scrub React 
// ParallaxImage.tsx
import { useEffect, useRef } from "react";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";

gsap.registerPlugin(ScrollTrigger);

interface ParallaxImageProps {
  src: string;
  alt: string;
  speed?: number; // parallax strength, e.g. 0.3 = 30% offset
  className?: string;
}

export function ParallaxImage({
  src,
  alt,
  speed = 0.3,
  className = "",
}: ParallaxImageProps) {
  const wrapRef = useRef<HTMLDivElement>(null);
  const imgRef = useRef<HTMLImageElement>(null);

  useEffect(() => {
    const wrap = wrapRef.current;
    const img = imgRef.current;
    if (!wrap || !img) return;

    // Offset = speed * 100% — image must be taller than container to cover gap
    const yPercent = speed * 100;

    const ctx = gsap.context(() => {
      gsap.fromTo(
        img,
        { yPercent: -yPercent },
        {
          yPercent,
          ease: "none",
          scrollTrigger: {
            trigger: wrap,
            start: "top bottom",
            end: "bottom top",
            scrub: true,
          },
        }
      );
    }, wrap);

    return () => ctx.revert();
  }, [speed]);

  const scale = 1 + speed * 2; // oversized so parallax doesn't reveal edges

  return (
    <div
      ref={wrapRef}
      className={`relative overflow-hidden ${className}`}
    >
      <img
        ref={imgRef}
        src={src}
        alt={alt}
        className="w-full h-full object-cover"
        style={{
          transform: `scale(${scale})`,
          willChange: "transform",
        }}
      />
    </div>
  );
}

9. Stagger Grid Reveal

Grid items animate in with a staggered delay cascade. Uses Intersection Observer + CSS variables for zero-dependency approach, plus a GSAP version.

Intersection Observer CSS stagger React 
// StaggerGrid.tsx
import { useEffect, useRef, ReactNode } from "react";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";

gsap.registerPlugin(ScrollTrigger);

interface StaggerGridProps {
  items: ReactNode[];
  columns?: number;
  stagger?: number;
  className?: string;
  itemClassName?: string;
}

export function StaggerGrid({
  items,
  columns = 3,
  stagger = 0.08,
  className = "",
  itemClassName = "",
}: StaggerGridProps) {
  const gridRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const grid = gridRef.current;
    if (!grid) return;

    const cards = grid.querySelectorAll<HTMLDivElement>(".stagger-item");

    const ctx = gsap.context(() => {
      gsap.fromTo(
        cards,
        {
          opacity: 0,
          y: 40,
          scale: 0.96,
        },
        {
          opacity: 1,
          y: 0,
          scale: 1,
          duration: 0.65,
          ease: "power3.out",
          stagger: {
            amount: stagger * items.length,
            grid: [Math.ceil(items.length / columns), columns],
            from: "start", // or "center", "edges", "random"
          },
          scrollTrigger: {
            trigger: grid,
            start: "top 80%",
            toggleActions: "play none none reverse",
          },
        }
      );
    }, grid);

    return () => ctx.revert();
  }, [items.length, columns, stagger]);

  const colClass: Record<number, string> = {
    2: "grid-cols-2",
    3: "grid-cols-3",
    4: "grid-cols-4",
  };

  return (
    <div
      ref={gridRef}
      className={`grid gap-6 ${colClass[columns] ?? "grid-cols-3"} ${className}`}
    >
      {items.map((item, i) => (
        <div
          key={i}
          className={`stagger-item ${itemClassName}`}
          style={{ willChange: "transform, opacity" }}
        >
          {item}
        </div>
      ))}
    </div>
  );
}

10. Text Scramble Effect

Random characters cycle through before settling on the final text. Classic hacker-style decode animation. Pure React hook, no dependencies.

React hook No deps Vanilla JS 
// useTextScramble.ts
import { useState, useEffect, useCallback, useRef } from "react";

const CHARS = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*";

function randomChar() {
  return CHARS[Math.floor(Math.random() * CHARS.length)];
}

export function useTextScramble(finalText: string, speed = 40) {
  const [displayed, setDisplayed] = useState(finalText);
  const frameRef = useRef<ReturnType<typeof setTimeout>>();
  const iterRef = useRef(0);

  const scramble = useCallback(() => {
    iterRef.current = 0;

    const totalFrames = finalText.length * 3; // how many frames before each char settles

    const tick = () => {
      setDisplayed(
        finalText
          .split("")
          .map((char, i) => {
            if (char === " ") return " ";
            // Once enough frames have passed for this char index, lock it in
            if (i < iterRef.current / 3) return char;
            return randomChar();
          })
          .join("")
      );

      iterRef.current++;

      if (iterRef.current < totalFrames + finalText.length) {
        frameRef.current = setTimeout(tick, speed);
      } else {
        setDisplayed(finalText);
      }
    };

    tick();
  }, [finalText, speed]);

  useEffect(() => {
    scramble();
    return () => clearTimeout(frameRef.current);
  }, [scramble]);

  return { displayed, scramble };
}

import { useTextScramble } from "./useTextScramble";

interface ScrambleTextProps {
  text: string;
  triggerOnHover?: boolean;
  className?: string;
  speed?: number;
}

export function ScrambleText({
  text,
  triggerOnHover = false,
  className = "",
  speed = 35,
}: ScrambleTextProps) {
  const { displayed, scramble } = useTextScramble(text, speed);

  return (
    <span
      className={`font-mono ${className}`}
      onMouseEnter={triggerOnHover ? scramble : undefined}
    >
      {displayed}
    </span>
  );
}

// ----------------------------------------------------------
// On-scroll trigger version
// ----------------------------------------------------------
export function ScrambleOnScroll({ text, className = "" }: { text: string; className?: string }) {
  const ref = useRef<HTMLSpanElement>(null);
  const { displayed, scramble } = useTextScramble(text, 35);
  const fired = useRef(false);

  useEffect(() => {
    const el = ref.current;
    if (!el) return;

    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting && !fired.current) {
          fired.current = true;
          scramble();
        }
      },
      { threshold: 0.5 }
    );

    observer.observe(el);
    return () => observer.disconnect();
  }, [scramble]);

  return (
    <span ref={ref} className={`font-mono ${className}`}>
      {displayed}
    </span>
  );
}
