Framer Motion — Advanced Patterns

Motion for React (formerly Framer Motion) — comprehensive API reference for useScroll, useTransform, useSpring, AnimatePresence, layout animations, drag, and production patterns.

npm i motion
import { motion, useScroll, useTransform } from 'motion/react'
Package rename: framer-motion was renamed to motion. Use import { ... } from 'motion/react' for React. The old package still works but motion is canonical.

Full API Quick Reference

Hooks

  • useScroll() — scroll position
  • useTransform() — map/compute motion values
  • useSpring() — spring physics
  • useMotionValue() — mutable animation value
  • useVelocity() — motion value velocity
  • useMotionTemplate() — CSS string interpolation
  • useMotionValueEvent() — listen to value changes
  • useAnimate() — imperative animations
  • useAnimationFrame() — rAF callback
  • useDragControls() — programmatic drag
  • useInView() — viewport detection
  • useReducedMotion() — a11y preference
  • useTime() — elapsed time value

Components

  • <motion.div> — animatable div
  • <AnimatePresence> — exit animations
  • <LayoutGroup> — sync layout anims
  • <LazyMotion> — reduce bundle size
  • <MotionConfig> — global config
  • <Reorder.Group> — drag-to-reorder
  • <AnimateView> — viewport-triggered

Key Props

  • initial / animate / exit
  • transition — timing/physics config
  • whileHover / whileTap / whileFocus
  • whileInView / viewport
  • layout / layoutId
  • drag / dragConstraints
  • variants / animate (variant name)

useScroll — Scroll-Linked Animations

Returns four motion values linked to scroll position. Hardware-accelerated via ScrollTimeline API when driving transform/opacity/filter/clipPath.

Return Values

ValueTypeDescription
scrollXMotionValue<number>Absolute horizontal scroll in px
scrollYMotionValue<number>Absolute vertical scroll in px
scrollXProgressMotionValue<0–1>Normalized horizontal progress
scrollYProgressMotionValue<0–1>Normalized vertical progress

Options

OptionDefaultDescription
containerviewportref to scrollable container
targetscrollable arearef to element being tracked
axis"y""x" | "y"
offset["start start","end end"]Define start/end intersection points
trackContentSizefalseAuto-update when content changes size

Element Scroll Progress (track specific element)

import { useRef } from 'react';
import { motion, useScroll, useTransform } from 'motion/react';

export function ParallaxSection() {
  const sectionRef = useRef<HTMLDivElement>(null);

  // Track progress of this specific element through viewport
  const { scrollYProgress } = useScroll({
    target: sectionRef,
    offset: ['start end', 'end start'],
    // "start end"   = element top hits viewport bottom (enters)
    // "end start"   = element bottom hits viewport top (exits)
  });

  // As element scrolls through, image moves up (parallax)
  const y = useTransform(scrollYProgress, [0, 1], ['0%', '-20%']);
  const opacity = useTransform(scrollYProgress, [0, 0.3, 0.7, 1], [0, 1, 1, 0]);

  return (
    <div ref={sectionRef} style={{ height: '80vh', overflow: 'hidden', position: 'relative' }}>
      <motion.img
        src="/hero.jpg"
        alt=""
        style={{ width: '100%', height: '120%', objectFit: 'cover', y, opacity }}
      />
    </div>
  );
}

useTransform — Map & Compute Motion Values

Creates a derived motion value. Two usage patterns: range mapping (input range → output range) and transform function (compute from one or more values).

import { useScroll, useTransform, motion } from 'motion/react';

// Pattern 1: Range mapping
// scrollYProgress: 0 → 1
// y:               0px → -100px
const { scrollYProgress } = useScroll();
const y = useTransform(scrollYProgress, [0, 1], [0, -100]);
const opacity = useTransform(scrollYProgress, [0, 0.5, 1], [1, 1, 0]);
const scale = useTransform(scrollYProgress, [0, 0.3], [1, 0.85]);

// Pattern 2: Transform function
// Reads from multiple motion values
const background = useTransform(
  () => `hsl(${scrollYProgress.get() * 360}, 100%, 50%)`
);

// Colors
const color = useTransform(
  scrollYProgress,
  [0, 1],
  ['#ffffff', '#f59e0b']
);

// CSS units
const blur = useTransform(scrollYProgress, [0, 1], ['blur(0px)', 'blur(8px)']);

// Options
const clamped = useTransform(
  scrollYProgress,
  [0, 0.5],
  [0, 1],
  { clamp: false }  // allow output beyond 0-1
);

// Multiple outputs from one input
const transforms = useTransform(scrollYProgress, [0, 1], {
  x: [0, 200],
  opacity: [1, 0],
  scale: [1, 0.5],
});

Parallax Stack (multiple layers)

import { useRef } from 'react';
import { motion, useScroll, useTransform } from 'motion/react';

export function ParallaxStack() {
  const ref = useRef<HTMLDivElement>(null);
  const { scrollYProgress } = useScroll({
    target: ref,
    offset: ['start start', 'end start'],
  });

  // Each layer moves at different speed
  const bgY = useTransform(scrollYProgress, [0, 1], ['0%', '30%']);     // slow
  const midY = useTransform(scrollYProgress, [0, 1], ['0%', '15%']);    // medium
  const fgY = useTransform(scrollYProgress, [0, 1], ['0%', '-10%']);    // forward
  const titleY = useTransform(scrollYProgress, [0, 1], ['0%', '60%']);  // fast

  return (
    <div ref={ref} style={{ height: '100vh', position: 'relative', overflow: 'hidden' }}>
      {/* Background layer */}
      <motion.div style={{ position: 'absolute', inset: 0, y: bgY,
        background: 'radial-gradient(circle at 50% 50%, #1a0a2e, #0a0a0a)' }} />

      {/* Mid layer */}
      <motion.img src="/mountains-mid.png" alt="" style={{
        position: 'absolute', bottom: 0, width: '100%', y: midY,
      }} />

      {/* Hero text */}
      <motion.h1 style={{
        position: 'absolute', top: '40%', left: '50%', transform: 'translateX(-50%)',
        y: titleY, color: '#fff', fontSize: '5rem', fontWeight: 700,
      }}>
        Parallax Hero
      </motion.h1>

      {/* Foreground layer */}
      <motion.img src="/mountains-fg.png" alt="" style={{
        position: 'absolute', bottom: 0, width: '100%', y: fgY,
      }} />
    </div>
  );
}

Color Theme on Scroll

import { useScroll, useTransform, motion, useMotionTemplate } from 'motion/react';

export function ThemeScroll() {
  const { scrollYProgress } = useScroll();

  const r = useTransform(scrollYProgress, [0, 0.5, 1], [10, 30, 15]);
  const g = useTransform(scrollYProgress, [0, 0.5, 1], [10, 15, 46]);
  const b = useTransform(scrollYProgress, [0, 0.5, 1], [10, 60, 22]);

  // useMotionTemplate stitches motion values into a CSS string
  const background = useMotionTemplate`rgb(${r}, ${g}, ${b})`;

  return <motion.div style={{ background, minHeight: '200vh', padding: 48 }} />;
}

useSpring — Spring Physics

Creates a motion value that animates with spring physics when its target changes. Can track a static value, a number, or another motion value (e.g. scroll progress for a smoothed lag).

import { useSpring, useScroll, motion } from 'motion/react';

// Spring configuration presets
const SPRING_CONFIGS = {
  gentle:   { stiffness: 100, damping: 20, mass: 1 },
  wobbly:   { stiffness: 180, damping: 12, mass: 1 },
  stiff:    { stiffness: 400, damping: 30, mass: 1 },
  slow:     { stiffness: 60,  damping: 15, mass: 1 },
  snappy:   { stiffness: 600, damping: 40, mass: 0.8 },
};

// Basic spring value
const scale = useSpring(1, SPRING_CONFIGS.wobbly);
// scale.set(1.2) → springs to 1.2 with physics

// Smoothed scroll progress
function SmoothedScrollBar() {
  const { scrollYProgress } = useScroll();
  // Spring follows scrollYProgress but with lag
  const smoothProgress = useSpring(scrollYProgress, {
    stiffness: 100,
    damping: 30,
    restDelta: 0.001,
    skipInitialAnimation: true,  // jump to initial value without animating
  });

  return <motion.div style={{ scaleX: smoothProgress, transformOrigin: '0%',
    height: 3, background: '#f59e0b', position: 'fixed', top: 0 }} />;
}

// Cursor follower with spring lag
function MagneticCursor() {
  const x = useSpring(0, { stiffness: 300, damping: 25 });
  const y = useSpring(0, { stiffness: 300, damping: 25 });

  const handleMouseMove = (e: React.MouseEvent) => {
    x.set(e.clientX - 12);
    y.set(e.clientY - 12);
  };

  return (
    <div onMouseMove={handleMouseMove} style={{ position: 'fixed', inset: 0, zIndex: 9999, pointerEvents: 'none' }}>
      <motion.div
        style={{
          position: 'fixed',
          x, y,
          width: 24,
          height: 24,
          borderRadius: '50%',
          background: 'rgba(245, 158, 11, 0.8)',
          pointerEvents: 'none',
        }}
      />
    </div>
  );
}

Interactive Spring Button

import { useSpring, motion } from 'motion/react';

export function SpringButton({ children }: { children: React.ReactNode }) {
  const scale = useSpring(1, { stiffness: 400, damping: 17 });
  const y = useSpring(0, { stiffness: 400, damping: 17 });

  return (
    <motion.button
      style={{ scale, y }}
      onHoverStart={() => { scale.set(1.05); y.set(-2); }}
      onHoverEnd={() => { scale.set(1); y.set(0); }}
      onTapStart={() => { scale.set(0.97); y.set(1); }}
      onTap={() => { scale.set(1); y.set(0); }}
      onTapCancel={() => { scale.set(1); y.set(0); }}
    >
      {children}
    </motion.button>
  );
}

AnimatePresence — Exit Animations

Enables exit prop on motion components. When a child is removed from the React tree, it plays its exit animation before unmounting.

Props

PropValuesDescription
mode"sync" | "wait" | "popLayout"sync = overlap; wait = sequential; popLayout = layout reflowed
initialbooleanfalse = skip enter anim on first render
customanyData passed to exiting children via usePresenceData()
onExitComplete() => voidFires when all exits finish
propagatebooleanTrigger child exits when parent exits

Modal with Exit Animation

import { useState } from 'react';
import { AnimatePresence, motion } from 'motion/react';

const backdropVariants = {
  hidden: { opacity: 0 },
  visible: { opacity: 1 },
};

const modalVariants = {
  hidden: { opacity: 0, scale: 0.9, y: 20 },
  visible: { opacity: 1, scale: 1, y: 0, transition: { type: 'spring', stiffness: 300, damping: 24 } },
  exit: { opacity: 0, scale: 0.95, y: 10, transition: { duration: 0.15 } },
};

export function Modal() {
  const [open, setOpen] = useState(false);

  return (
    <>
      <button onClick={() => setOpen(true)}>Open Modal</button>

      <AnimatePresence mode="wait">
        {open && (
          <motion.div
            key="backdrop"
            variants={backdropVariants}
            initial="hidden"
            animate="visible"
            exit="hidden"
            onClick={() => setOpen(false)}
            style={{
              position: 'fixed', inset: 0, background: 'rgba(0,0,0,0.6)',
              display: 'flex', alignItems: 'center', justifyContent: 'center',
              zIndex: 100,
            }}
          >
            <motion.div
              key="modal"
              variants={modalVariants}
              initial="hidden"
              animate="visible"
              exit="exit"
              onClick={e => e.stopPropagation()}
              style={{
                background: '#111', borderRadius: 16, padding: 32,
                width: 480, border: '1px solid #1f1f1f',
              }}
            >
              <h2 style={{ color: '#fff' }}>Modal Title</h2>
              <button onClick={() => setOpen(false)}>Close</button>
            </motion.div>
          </motion.div>
        )}
      </AnimatePresence>
    </>
  );
}

Page Transitions (Next.js App Router)

// app/layout.tsx — wrap children in AnimatePresence
'use client';
import { AnimatePresence, motion } from 'motion/react';
import { usePathname } from 'next/navigation';

const pageVariants = {
  initial: { opacity: 0, y: 8 },
  animate: { opacity: 1, y: 0, transition: { duration: 0.3, ease: [0.22, 1, 0.36, 1] } },

            key={pathname}
            variants={pageVariants}
            initial="initial"
            animate="animate"
            exit="exit"
          >
            {children}
          </motion.main>
        </AnimatePresence>
      </body>
    </html>
  );
}

List with Enter/Exit Stagger

import { useState } from 'react';
import { AnimatePresence, motion } from 'motion/react';

const listVariants = {
  container: {
    animate: { transition: { staggerChildren: 0.07 } },
    exit: { transition: { staggerChildren: 0.05, staggerDirection: -1 } },
  },
  item: {
    initial: { opacity: 0, x: -20 },
    animate: { opacity: 1, x: 0 },
    exit: { opacity: 0, x: 20, transition: { duration: 0.2 } },
  },
};

export function AnimatedList() {
  const [items, setItems] = useState(['Alpha', 'Beta', 'Gamma', 'Delta']);

  const remove = (item: string) => setItems(prev => prev.filter(i => i !== item));
  const add = () => setItems(prev => [...prev, `Item ${Date.now()}`]);

  return (
    <div>
      <button onClick={add}>Add Item</button>
      <motion.ul
        variants={listVariants.container}
        initial="initial"
        animate="animate"
        exit="exit"
        style={{ listStyle: 'none', padding: 0 }}
      >
        <AnimatePresence>
          {items.map(item => (
            <motion.li
              key={item}
              variants={listVariants.item}
              layout
              style={{ padding: '8px 12px', marginBottom: 4, background: '#111', borderRadius: 8, cursor: 'pointer' }}
              onClick={() => remove(item)}
            >
              {item}
            </motion.li>
          ))}
        </AnimatePresence>
      </motion.ul>
    </div>
  );
}

Layout Animations & Shared Element Transitions

Add layout to any motion component to automatically animate its position/size changes. Use layoutId to connect two different elements for shared element transitions.

All Layout Props

PropDescription
layouttrue | "position" | "size" — auto-animate changes
layoutIdstring — link elements across components for shared transition
layoutScrollApply to scrollable containers inside layout animations
layoutRootApply to position:fixed elements to account for scroll
layoutAnchor{x: 0–1, y: 0–1} — anchor point during layout animation
layoutDependencyTrigger layout recalc on value change

Expanding Card (layoutId shared transition)

import { useState } from 'react';
import { AnimatePresence, motion, LayoutGroup } from 'motion/react';

const cards = [
  { id: 'card-1', title: 'Card One', color: '#f59e0b' },
  { id: 'card-2', title: 'Card Two', color: '#60a5fa' },
  { id: 'card-3', title: 'Card Three', color: '#34d399' },
];

export function ExpandingCards() {
  const [selected, setSelected] = useState<string | null>(null);

  return (
    <LayoutGroup>
      <div style={{ display: 'flex', gap: 16 }}>
        {cards.map(card => (
          <motion.div
            key={card.id}
            layoutId={card.id}   // same id as expanded version
            onClick={() => setSelected(card.id)}
            style={{
              width: 120, height: 80, borderRadius: 12, cursor: 'pointer',
              background: card.color,
            }}
          >
            <motion.h3 layoutId={`${card.id}-title`} style={{ padding: 12, fontSize: 14 }}>
              {card.title}
            </motion.h3>
          </motion.div>
        ))}
      </div>

      <AnimatePresence>
        {selected && (() => {
          const card = cards.find(c => c.id === selected)!;
          return (
            <motion.div
              layoutId={selected}
              onClick={() => setSelected(null)}
              style={{
                position: 'fixed', inset: '10%', zIndex: 100,
                background: card.color, borderRadius: 24,
                cursor: 'pointer', padding: 32,
              }}
            >
              <motion.h2 layoutId={`${selected}-title`} style={{ fontSize: 28 }}>
                {card.title}
              </motion.h2>
              <motion.p
                initial={{ opacity: 0 }}
                animate={{ opacity: 1, transition: { delay: 0.2 } }}
                exit={{ opacity: 0 }}
              >
                Expanded content here...
              </motion.p>
            </motion.div>
          );
        })()}
      </AnimatePresence>
    </LayoutGroup>
  );
}

Variants — Orchestration & Stagger

Variants allow you to define animation states by name and propagate them through component trees. Parent controls children via stagger.

import { motion } from 'motion/react';

const containerVariants = {
  hidden: { opacity: 0 },
  visible: {
    opacity: 1,
    transition: {
      // Orchestration options
      staggerChildren: 0.1,       // delay between each child
      delayChildren: 0.2,         // initial delay before first child
      staggerDirection: 1,        // 1 = forward, -1 = reverse
      when: 'beforeChildren',     // 'beforeChildren' | 'afterChildren'
    },
  },
};

const itemVariants = {
  hidden: { opacity: 0, y: 20 },
  visible: {
    opacity: 1,
    y: 0,
    transition: { type: 'spring', stiffness: 300, damping: 20 },
  },
};

export function StaggerGrid() {
  const items = Array.from({ length: 9 }, (_, i) => i);

  return (
    <motion.div
      variants={containerVariants}
      initial="hidden"
      whileInView="visible"
      viewport={{ once: true, margin: '-50px' }}
      style={{ display: 'grid', gridTemplateColumns: 'repeat(3,1fr)', gap: 12 }}
    >
      {items.map(i => (
        <motion.div
          key={i}
          variants={itemVariants}
          style={{
            background: '#111', borderRadius: 8,
            height: 100, border: '1px solid #1f1f1f',
          }}
        />
      ))}
    </motion.div>
  );
}

  hover: { rotate: 15, x: 3, transition: { type: 'spring', stiffness: 400 } },
};

export function Card3DButton({ icon, title }: { icon: string; title: string }) {
  return (
    <motion.button
      variants={cardVariants}
      initial="rest"
      whileHover="hover"
      whileTap="tap"
      style={{
        background: '#111', border: '1px solid #1f1f1f',
        borderRadius: 12, padding: '20px 24px', cursor: 'pointer',
        display: 'flex', alignItems: 'center', gap: 12, color: '#fff',
      }}
    >
      {/* Child inherits parent's "hover" variant automatically */}
      <motion.span variants={iconVariants} style={{ fontSize: 24 }}>
        {icon}
      </motion.span>
      <span style={{ fontWeight: 600 }}>{title}</span>
    </motion.button>
  );
}

Theatre.js — Professional Animation Sequencer

Theatre.js is a JavaScript animation library with a visual motion design editor. Animate any JS variable — Three.js, R3F, CSS, DOM — using a timeline GUI. Free and open source.

Packages

npm i @theatre/core @theatre/studio
npm i @theatre/r3f # React Three Fiber
npm i @theatre/react # React bindings

Core Concepts

  • Project — container for all animations
  • Sheet — timeline/scene group
  • Sheet Object — animatable entity
  • Sequence — keyframe timeline
  • Prop types — number, boolean, string, rgba

Basic Theatre.js Setup

import { getProject, types } from '@theatre/core';
import studio from '@theatre/studio';

// Initialize studio (dev only — remove in production)
if (process.env.NODE_ENV === 'development') {
  studio.initialize();
}

// Create a project
const project = getProject('My Animation Project');

// Create a sheet (like a scene/timeline)
const sheet = project.sheet('Hero Scene');

// Create an animatable object with typed props
const obj = sheet.object('Camera', {
  position: types.compound({
    x: types.number(0, { range: [-20, 20] }),
    y: types.number(0, { range: [-10, 10] }),
    z: types.number(5, { range: [0, 30] }),
  }),
  rotation: types.number(0, { range: [-180, 180] }),
  fov: types.number(60, { range: [20, 120] }),
});

// Subscribe to changes (runs on every frame during playback)
obj.onValuesChange((values) => {
  camera.position.set(values.position.x, values.position.y, values.position.z);
  camera.rotation.y = values.rotation * (Math.PI / 180);
  camera.fov = values.fov;
  camera.updateProjectionMatrix();
});

// Play the sequence
const sequence = sheet.sequence;
sequence.play({ iterationCount: Infinity, range: [0, 10] });

// Programmatic control
sequence.pause();
sequence.position = 3.5; // seek to 3.5 seconds

useAnimate — Imperative Animations

Programmatic control of animations without relying on component re-renders. Returns a scope ref and an animate function.

import { useAnimate, stagger } from 'motion/react';

export function AnimateOnClick() {
  const [scope, animate] = useAnimate();

  const runAnimation = async () => {
    // scope scopes selectors to this ref's subtree
    await animate('li', { opacity: [0, 1], y: [20, 0] }, {
      duration: 0.3,
      delay: stagger(0.05),
    });

    await animate('li', { color: '#f59e0b' }, { duration: 0.2 });
    await animate('.badge', { scale: [0, 1] }, { type: 'spring', stiffness: 400 });
  };

  return (
    <div ref={scope}>
      <button onClick={runAnimation}>Animate</button>
      <ul>
        {['Alpha', 'Beta', 'Gamma'].map(item => (
          <li key={item} style={{ opacity: 0 }}>
            {item} <span className="badge">new</span>
          </li>
        ))}
      </ul>
    </div>
  );
}

Performance & Accessibility Patterns

LazyMotion — Reduce Bundle Size

import { LazyMotion, domAnimation, m } from 'motion/react';

// domAnimation = ~18kb (vs full ~34kb)
// domMax = full feature set including drag

export function OptimizedApp({ children }: { children: React.ReactNode }) {
  return (
    <LazyMotion features={domAnimation} strict>
      {children}
    </LazyMotion>
  );
}

// Use <m.div> instead of <motion.div> inside LazyMotion
function Card() {
  return <m.div initial={{ opacity: 0 }} animate={{ opacity: 1 }}>Card</m.div>;
}

MotionConfig — Global Defaults

import { MotionConfig } from 'motion/react';
import { useReducedMotion } from 'motion/react';

export function AppMotionConfig({ children }: { children: React.ReactNode }) {
  const prefersReduced = useReducedMotion();

  return (
    <MotionConfig
      transition={{
        duration: prefersReduced ? 0 : 0.35,
        ease: [0.22, 1, 0.36, 1],  // custom cubic-bezier
      }}
      reducedMotion={prefersReduced ? 'always' : 'never'}
    >
      {children}
    </MotionConfig>
  );
}

whileInView — Scroll-Triggered (No ScrollTrigger needed)

import { motion } from 'motion/react';

export function RevealSection() {
  return (
    <motion.section
      initial={{ opacity: 0, y: 40 }}
      whileInView={{ opacity: 1, y: 0 }}
      viewport={{
        once: true,          // only animate once
        margin: '-100px',    // trigger 100px before entering viewport
        amount: 0.3,         // trigger when 30% is visible
      }}
      transition={{ duration: 0.6, ease: [0.22, 1, 0.36, 1] }}
    >
      Content that animates in on scroll
    </motion.section>
  );
}
