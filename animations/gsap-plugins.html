ScrollTrigger Free

The most popular GSAP plugin. Trigger animations when elements enter/leave the viewport. Supports pinning, scrubbing (linking progress to scroll position), snapping, and callbacks.

gsap.registerPlugin(ScrollTrigger)

Basic Trigger on Enter

import { gsap } from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';

gsap.registerPlugin(ScrollTrigger);

// Animate when .box enters the viewport
gsap.to('.box', {
  x: 500,
  opacity: 1,
  duration: 1,
  ease: 'power3.out',
  scrollTrigger: {
    trigger: '.box',      // element that triggers the animation
    start: 'top 80%',     // "element top" meets "viewport 80%"
    end: 'bottom 20%',    // "element bottom" meets "viewport 20%"
    toggleActions: 'play none none reverse',
    // toggleActions: 'onEnter onLeave onEnterBack onLeaveBack'
    // values: play | pause | resume | reset | restart | complete | reverse | none
  },
});

Scrub — Link to Scroll Progress

// scrub: true — instantly follows scroll
// scrub: 0.5 — 0.5s lag (smoother feel)
gsap.to('.parallax-img', {
  y: -200,
  scrollTrigger: {
    trigger: '.section',
    start: 'top top',
    end: 'bottom top',
    scrub: 1,
  },
});

Pin — Lock Element During Scroll

// Pin .hero in place while the page scrolls through 500px of content
ScrollTrigger.create({
  trigger: '.hero',
  start: 'top top',
  end: '+=500',         // pin for 500px of scroll distance
  pin: true,
  pinSpacing: true,     // adds space below pinned element
});

React Hook (reusable)

import { useEffect, useRef } from 'react';
import { gsap } from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';

gsap.registerPlugin(ScrollTrigger);

export function useScrollReveal<T extends HTMLElement>(options = {}) {
  const ref = useRef<T>(null);

  useEffect(() => {
    if (!ref.current) return;
    const el = ref.current;

    const ctx = gsap.context(() => {
      gsap.from(el, {
        opacity: 0,
        y: 40,
        duration: 0.7,
        ease: 'power2.out',
        scrollTrigger: {
          trigger: el,
          start: 'top 85%',
          toggleActions: 'play none none none',
          ...options,
        },
      });
    }, el);

    return () => ctx.revert();
  }, []);

  return ref;
}

// Usage:
// const ref = useScrollReveal<HTMLDivElement>({ start: 'top 70%' });
// <div ref={ref}>...</div>

ScrollSmoother Free

Smooth scrolling built on native scroll technology (not intercepted JS). Requires ScrollTrigger. Adds momentum, speed effects, and parallax to elements.

gsap.registerPlugin(ScrollTrigger, ScrollSmoother)
<!-- Required HTML structure -->
<body>
  <div id="smooth-wrapper">
    <div id="smooth-content">
      <!-- ALL your page content here -->
    </div>
  </div>
</body>
import { gsap } from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
import { ScrollSmoother } from 'gsap/ScrollSmoother';

gsap.registerPlugin(ScrollTrigger, ScrollSmoother);

const smoother = ScrollSmoother.create({
  wrapper: '#smooth-wrapper',
  content: '#smooth-content',
  smooth: 1.5,           // smoothing factor (0 = no smooth, 2 = very smooth)
  effects: true,         // enable data-speed and data-lag attributes
  normalizeScroll: true, // prevents janky mobile scroll
});

// Parallax via data attributes:
// <img data-speed="0.5" />   — moves at 50% scroll speed
// <div data-lag="0.3" />    — lags 0.3s behind scroll

// Programmatic scroll
smoother.scrollTo('#section-3', true, 'top top');

// Pause/resume
smoother.paused(true);
smoother.paused(false);

Observer Free

Unified event handler for wheel, touch, pointer, and scroll. Perfect for full-page scroll snapping without hijacking native scroll. If ScrollTrigger is loaded, access via ScrollTrigger.observe().

gsap.registerPlugin(Observer)
import { gsap } from 'gsap';
import { Observer } from 'gsap/Observer';

gsap.registerPlugin(Observer);

// Full-page section snapping
const sections = document.querySelectorAll('.section');
let current = 0;
let animating = false;

function goTo(index) {
  if (animating || index === current) return;
  animating = true;
  const prev = sections[current];
  const next = sections[index];
  const direction = index > current ? 1 : -1;

  gsap.timeline({
    onComplete: () => { animating = false; current = index; },
  })
    .to(prev, { yPercent: -100 * direction, duration: 1, ease: 'power2.inOut' }, 0)
    .from(next, { yPercent: 100 * direction, duration: 1, ease: 'power2.inOut' }, 0);
}

Observer.create({
  target: window,
  type: 'wheel,touch,pointer',
  wheelSpeed: -1,
  onDown: () => goTo(Math.min(current + 1, sections.length - 1)),
  onUp: () => goTo(Math.max(current - 1, 0)),
  tolerance: 10,
  preventDefault: true,
});

Flip Free

Animate between two DOM states using the FLIP (First, Last, Invert, Play) technique. Elements can move across layouts, grids, or containers seamlessly — even with DOM restructuring.

gsap.registerPlugin(Flip)
import { gsap } from 'gsap';
import { Flip } from 'gsap/Flip';

gsap.registerPlugin(Flip);

// 3-step pattern
function toggleLayout() {
  // Step 1: Record current state (positions, sizes)
  const state = Flip.getState('.item');

  // Step 2: Make DOM/class changes that alter layout
  document.querySelector('.container').classList.toggle('grid-view');

  // Step 3: Animate from old state to new state
  Flip.from(state, {
    duration: 0.6,
    ease: 'power1.inOut',
    stagger: 0.05,
    absolute: true,         // use position:absolute during animation
    onEnter: (elements) => gsap.from(elements, { opacity: 0, scale: 0, duration: 0.3 }),
    onLeave: (elements) => gsap.to(elements, { opacity: 0, scale: 0, duration: 0.3 }),
  });
}

React — Grid / List Toggle

import { useState, useLayoutEffect, useRef } from 'react';
import { gsap } from 'gsap';
import { Flip } from 'gsap/Flip';

gsap.registerPlugin(Flip);

export default function FlipGrid() {
  const [isGrid, setIsGrid] = useState(true);
  const containerRef = useRef<HTMLDivElement>(null);
  const stateRef = useRef<ReturnType<typeof Flip.getState> | null>(null);

  const toggle = () => {
    // Capture before the React re-render
    stateRef.current = Flip.getState('.flip-item');
    setIsGrid(v => !v);
  };

  useLayoutEffect(() => {
    if (!stateRef.current) return;
    Flip.from(stateRef.current, {
      duration: 0.5,
      ease: 'power2.inOut',
      stagger: 0.04,
      absolute: true,
    });
    stateRef.current = null;
  }, [isGrid]);

  const items = ['A','B','C','D','E','F'];

  return (
    <div>
      <button onClick={toggle}>Toggle Layout</button>
      <div
        ref={containerRef}
        style={{
          display: isGrid ? 'grid' : 'flex',
          gridTemplateColumns: isGrid ? 'repeat(3,1fr)' : undefined,
          flexDirection: isGrid ? undefined : 'column',
          gap: 12,
        }}
      >
        {items.map(i => (
          <div key={i} className="flip-item" style={{
            background: '#1a1a1a', padding: 24, borderRadius: 8, color: '#fff',
          }}>
            Item {i}
          </div>
        ))}
      </div>
    </div>
  );
}

SplitText Free

Split HTML text into individual characters, words, and/or lines — each wrapped in a span — for precise stagger animations. Automatically handles responsive re-splitting and accessibility (aria attributes).

gsap.registerPlugin(SplitText)
import { gsap } from 'gsap';
import { SplitText } from 'gsap/SplitText';

gsap.registerPlugin(SplitText);

// Split by chars and words
const split = SplitText.create('.headline', {
  type: 'chars,words,lines',
  linesClass: 'line',       // CSS class applied to line wrappers
  charsClass: 'char',
  wordsClass: 'word',
  aria: 'description',      // accessibility: keep original text for screen readers
});

// Staggered character reveal from below
gsap.from(split.chars, {
  duration: 0.6,
  y: 100,
  autoAlpha: 0,
  stagger: 0.03,
  ease: 'back.out(1.7)',
  scrollTrigger: {
    trigger: '.headline',
    start: 'top 80%',
  },
});

// Cleanup
split.revert();   // restores original HTML

DrawSVG Free

Progressively draw or erase SVG strokes by manipulating stroke-dashoffset and stroke-dasharray. Works with paths, circles, polylines, etc.

gsap.registerPlugin(DrawSVGPlugin)
<!-- SVG must have stroke defined -->
<svg viewBox="0 0 200 200">
  <circle class="ring" cx="100" cy="100" r="80"
    fill="none" stroke="#f59e0b" stroke-width="4" />
  <path class="logo-path" d="M10 80 Q 95 10 180 80 Q 95 150 10 80"
    fill="none" stroke="#fff" stroke-width="2" />
</svg>
import { gsap } from 'gsap';
import { DrawSVGPlugin } from 'gsap/DrawSVGPlugin';

gsap.registerPlugin(DrawSVGPlugin);

// Draw from 0 to full
gsap.from('.ring', { drawSVG: 0, duration: 2, ease: 'power2.inOut' });

// Draw from center outward
gsap.from('.ring', { drawSVG: '50% 50%', duration: 1.5 });

// Erase (reverse direction)
gsap.to('.logo-path', { drawSVG: 0, duration: 1.5 });

// Partial draw (draw first 40%)
gsap.from('.logo-path', { drawSVG: '0% 40%', duration: 1 });

// Timeline sequence: draw then erase
gsap.timeline({ scrollTrigger: { trigger: 'svg', start: 'top center' } })
  .from('.logo-path', { drawSVG: 0, duration: 2 })
  .to('.logo-path', { drawSVG: 0, duration: 1.5, delay: 0.5 });

MotionPathPlugin Free

Animate any element along an SVG path with auto-rotation. Supports path data strings, SVG elements, and coordinate arrays.

gsap.registerPlugin(MotionPathPlugin)
<svg viewBox="0 0 500 300">
  <path id="track"
    d="M10,150 C100,50 200,250 300,150 S450,50 490,150"
    fill="none" stroke="#333" stroke-width="1" />
  <div id="car">🚀</div>
</svg>
import { gsap } from 'gsap';
import { MotionPathPlugin } from 'gsap/MotionPathPlugin';

gsap.registerPlugin(MotionPathPlugin);

gsap.to('#car', {
  motionPath: {
    path: '#track',          // SVG path element selector
    align: '#track',         // align element to path coordinate system
    alignOrigin: [0.5, 0.5], // anchor point of the element (center)
    autoRotate: true,        // rotate element to face direction of motion
    start: 0,                // 0 = path start
    end: 1,                  // 1 = path end
  },
  duration: 4,
  ease: 'none',
  repeat: -1,
});

// Using path data string directly (no SVG element needed)
gsap.to('.dot', {
  motionPath: {
    path: 'M0,0 C100,-100 200,100 300,0',
    autoRotate: 90,    // offset rotation by 90°
  },
  duration: 3,
  ease: 'power1.inOut',
});

TextPlugin Free

Animate text replacement character-by-character or word-by-word. Useful for typewriter effects and dynamic content transitions.

gsap.registerPlugin(TextPlugin)
import { gsap } from 'gsap';
import { TextPlugin } from 'gsap/TextPlugin';

gsap.registerPlugin(TextPlugin);

// Basic typewriter
gsap.to('.headline', {
  text: 'Hello, World!',
  duration: 2,
  ease: 'none',   // linear for typewriter feel
});

// Word by word
gsap.to('.subtitle', {
  text: { value: 'Fast beautiful production ready', delimiter: ' ' },
  duration: 1.5,
  ease: 'none',
});

// Cycle through phrases (looping typewriter)
const phrases = ['Build faster.', 'Ship better.', 'Look great.'];
let i = 0;

function cycleText() {
  gsap.timeline({ onComplete: cycleText })
    .to('.hero-text', {
      text: phrases[i % phrases.length],
      duration: 1.5,
      ease: 'none',
    })
    .to('.hero-text', { opacity: 0.3, duration: 0.3, delay: 1 })
    .to('.hero-text', { opacity: 1, duration: 0.2 });
  i++;
}
cycleText();

GSAP + React — Best Practices

Always use gsap.context() with a cleanup function to prevent memory leaks in React components.

import { useRef, useLayoutEffect } from 'react';
import { gsap } from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
import { SplitText } from 'gsap/SplitText';

gsap.registerPlugin(ScrollTrigger, SplitText);

export default function AnimatedSection() {
  const sectionRef = useRef<HTMLElement>(null);

  useLayoutEffect(() => {
    // gsap.context scopes all selectors to sectionRef
    const ctx = gsap.context(() => {
      const split = SplitText.create('.section-title', { type: 'chars' });

      gsap.from(split.chars, {
        opacity: 0,
        y: 40,
        stagger: 0.04,
        duration: 0.5,
        ease: 'power2.out',
        scrollTrigger: {
          trigger: '.section-title',
          start: 'top 80%',
        },
      });

      gsap.from('.section-body', {
        opacity: 0,
        y: 24,
        duration: 0.6,
        delay: 0.3,
        scrollTrigger: {
          trigger: '.section-body',
          start: 'top 85%',
        },
      });
    }, sectionRef); // <-- scope

    return () => ctx.revert(); // <-- cleanup on unmount
  }, []);

  return (
    <section ref={sectionRef}>
      <h2 className="section-title">Animated Title</h2>
      <p className="section-body">This paragraph fades in on scroll.</p>
    </section>
  );
}
