React Animation Library

Framer Motion

The most popular React animation library. Declarative animations with a simple props API, gesture support, layout animations, scroll-linked effects, and AnimatePresence for mount/unmount transitions. Powers Framer's visual editor.

Installation

bash
npm install framer-motion
# or: yarn add framer-motion  |  pnpm add framer-motion
Note: In Next.js App Router, import from 'framer-motion' works in client components. Use the 'use client' directive. For server components, wrap animated content in a client component boundary.

motion.* Component

Any HTML or SVG element can be made animatable by prefixing with "motion." — e.g. motion.div, motion.section, motion.svg.

jsx
import { motion } from 'framer-motion';

// Basic animate prop
<motion.div
  animate={{ x: 100, opacity: 1 }}
  initial={{ x: 0, opacity: 0 }}
  exit={{ x: -100, opacity: 0 }}
  transition={{ duration: 0.5, ease: 'easeOut' }}
/>

// Mount animation (initial to animate)
<motion.div
  initial={{ opacity: 0, y: 20 }}
  animate={{ opacity: 1, y: 0 }}
  transition={{ duration: 0.6, ease: [0.22, 1, 0.36, 1] }}
/>

// All animatable properties
<motion.div
  animate={{
    x: 100,             // translateX (px)
    y: -50,             // translateY (px)
    rotate: 360,        // degrees
    scale: 1.5,
    scaleX: 2,
    opacity: 0.8,
    backgroundColor: '#f59e0b',
    borderRadius: '50%',
    width: '200px',
    pathLength: 1,      // SVG path drawing
    filter: 'blur(0px)'
  }}
/>

// Keyframes (array syntax)
<motion.div
  animate={{
    x: [0, 100, 50, 100],
    opacity: [0, 1, 0.5, 1],
  }}
  transition={{
    duration: 2,
    times: [0, 0.3, 0.6, 1],  // progress points (0-1) for each keyframe
    ease: ['easeIn', 'easeOut', 'linear']
  }}
/>

Variants

Variants let you define named animation states and orchestrate parent-child animations. The most powerful pattern in Framer Motion.

jsx
// Define variants as objects
const containerVariants = {
  hidden: { opacity: 0 },
  visible: {
    opacity: 1,
    transition: {
      staggerChildren: 0.1,   // stagger child animations by 0.1s
      delayChildren: 0.2,     // wait 0.2s before starting children
      when: 'beforeChildren',  // 'beforeChildren' | 'afterChildren'
    }
  },
  exit: { opacity: 0, transition: { staggerChildren: 0.05, staggerDirection: -1 } }
};

const itemVariants = {
  hidden: { opacity: 0, y: 20 },
  visible: {
    opacity: 1,
    y: 0,
    transition: { duration: 0.5, ease: [0.22, 1, 0.36, 1] }
  },
  exit: { opacity: 0, y: -10 }
};

// Parent automatically propagates variant names to children
function AnimatedList({ items }) {
  return (
    <motion.ul
      variants={containerVariants}
      initial="hidden"
      animate="visible"
      exit="exit"
    >
      {items.map(item => (
        <motion.li key={item.id} variants={itemVariants}>
          {item.text}
        </motion.li>
      ))}
    </motion.ul>
  );
}

// whileHover / whileTap / whileFocus
<motion.button
  variants={buttonVariants}
  whileHover="hover"
  whileTap="tap"
  whileFocus="focus"
/>

AnimatePresence

Enables exit animations when components unmount. Wrap conditional renders or route changes with AnimatePresence.

jsx
import { AnimatePresence, motion } from 'framer-motion';

// Toggle component visibility
function Modal({ isOpen, onClose }) {
  return (
    <AnimatePresence>
      {isOpen && (
        <motion.div
          className="modal-overlay"
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          onClick={onClose}
        >
          <motion.div
            className="modal"
            initial={{ scale: 0.9, opacity: 0, y: 20 }}
            animate={{ scale: 1, opacity: 1, y: 0 }}
            exit={{ scale: 0.95, opacity: 0, y: 10 }}
            transition={{ type: 'spring', stiffness: 300, damping: 30 }}
            onClick={e => e.stopPropagation()}
          >
            {/* content */}
          </motion.div>
        </motion.div>
      )}
    </AnimatePresence>
  );
}

// Page transitions (Next.js App Router)
// app/template.jsx — template re-renders on every navigation
'use client';
export default function Template({ children }) {
  return (
    <AnimatePresence mode="wait">
      <motion.div
        initial={{ opacity: 0, y: 16 }}
        animate={{ opacity: 1, y: 0 }}
        exit={{ opacity: 0, y: -8 }}
        transition={{ duration: 0.3, ease: 'easeInOut' }}
      >
        {children}
      </motion.div>
    </AnimatePresence>
  );
}

// AnimatePresence mode prop
// "sync"     — enter and exit at same time (default)
// "wait"     — wait for exit to finish before entering
// "popLayout" — animate out element is popped from layout flow

Gestures

Framer Motion has built-in gesture support with while* props and event handlers.

jsx
// Hover and tap gestures
<motion.button
  whileHover={{ scale: 1.05, backgroundColor: '#f59e0b' }}
  whileTap={{ scale: 0.95 }}
  whileFocus={{ outline: '2px solid #f59e0b' }}
  transition={{ type: 'spring', stiffness: 400, damping: 20 }}
>
  Click me
</motion.button>

// Drag
<motion.div
  drag                           // drag on both axes
  drag="x"                       // or constrain to x axis
  dragConstraints={{ left: -100, right: 100, top: 0, bottom: 0 }}
  dragElastic={0.2}              // bounciness at constraints (0 = rigid)
  dragMomentum={true}            // throw momentum after release
  onDragStart={(event, info) => console.log(info.point)}
  onDrag={(event, info) => console.log(info.offset)}
  onDragEnd={(event, info) => console.log(info.velocity)}
  whileDrag={{ scale: 1.1, cursor: 'grabbing' }}
/>

// Constrain drag to a container ref
const constraintsRef = useRef(null);
<div ref={constraintsRef} className="container">
  <motion.div drag dragConstraints={constraintsRef} />
</div>

// Pan events
<motion.div
  onPanStart={(e, info) => {}}
  onPan={(e, info) => console.log(info.delta, info.offset, info.velocity)}
  onPanEnd={(e, info) => {}}
/>

Layout Animations

The layout prop automatically animates an element when its layout changes — no manual calculations needed.

jsx
// Animate layout changes automatically
<motion.div layout> ... </motion.div>

// layout="position" — only animate x/y, not size
// layout="size"     — only animate size, not position
// layout="preserve-aspect" — maintain aspect ratio during resize

// Shared layout — element moves between positions across components
// Use layoutId to link elements
function TabIndicator({ activeTab }) {
  return tabs.map(tab => (
    <button key={tab.id} onClick={() => setActiveTab(tab.id)}>
      {tab.label}
      {activeTab === tab.id && (
        <motion.div
          layoutId="tab-indicator"  // must match across positions
          className="underline"
          transition={{ type: 'spring', stiffness: 380, damping: 30 }}
        />
      )}
    </button>
  ));
}

// LayoutGroup — needed when layoutId elements are in different components
import { LayoutGroup } from 'framer-motion';
<LayoutGroup>
  <ComponentA />
  <ComponentB />
</LayoutGroup>

useScroll & useTransform

useScroll tracks scroll position. useTransform maps scroll values to animation values — the Framer Motion way to do parallax and scroll-linked animations.

jsx
import { useScroll, useTransform, useSpring, motion } from 'framer-motion';
import { useRef } from 'react';

// Global scroll progress
function ScrollProgressBar() {
  const { scrollYProgress } = useScroll();
  return <motion.div className="progress-bar" style={{ scaleX: scrollYProgress }} />;
}

// Element-relative scroll
function ParallaxSection() {
  const ref = useRef(null);
  const { scrollYProgress } = useScroll({
    target: ref,
    offset: ['start end', 'end start']  // [when element enters, when element leaves]
    // offset options: "start" | "center" | "end" | "0.8" | px value
  });

  // Map scroll 0-1 to pixel values
  const y = useTransform(scrollYProgress, [0, 1], [0, -100]);
  const opacity = useTransform(scrollYProgress, [0, 0.3, 0.7, 1], [0, 1, 1, 0]);
  const scale = useTransform(scrollYProgress, [0, 0.5], [0.8, 1]);

  // Smooth the motion value with spring
  const smoothY = useSpring(y, { stiffness: 100, damping: 30 });

  return (
    <section ref={ref}>
      <motion.div style={{ y: smoothY, opacity, scale }}>
        Parallax content
      </motion.div>
    </section>
  );
}

// useScrollVelocity — scroll speed
import { useScrollVelocity } from 'framer-motion';
const velocity = useScrollVelocity(scrollYProgress);
const skewX = useTransform(velocity, [-2000, 0, 2000], [-15, 0, 15]);

Spring & Transition Config

Framer Motion has two animation engines: spring (physics-based) and tween (duration-based). Spring is the default.

jsx
// Spring (physics-based — no duration)
transition={{
  type: 'spring',
  stiffness: 300,    // higher = faster/stiffer
  damping: 20,       // higher = less bounce
  mass: 1,           // higher = more inertia
  velocity: 0,       // initial velocity
  restDelta: 0.001,  // when to consider animation done
  restSpeed: 0.001
}}

// Common spring presets
// Snappy: stiffness:400, damping:25
// Bouncy: stiffness:200, damping:10
// Gentle: stiffness:100, damping:20
// Stiff:  stiffness:700, damping:35

// Tween (duration-based)
transition={{
  type: 'tween',     // or just omit type and set duration
  duration: 0.5,
  ease: 'easeOut',
  // ease can be: 'linear' | 'easeIn' | 'easeOut' | 'easeInOut'
  // or cubic bezier: [0.22, 1, 0.36, 1]
  // or named: 'anticipate' | 'backIn' | 'backOut' | 'backInOut' | 'circIn' etc
  delay: 0.2,
  repeat: Infinity,
  repeatType: 'reverse', // 'loop' | 'reverse' | 'mirror'
  repeatDelay: 1
}}

// Per-property transitions
<motion.div
  animate={{ x: 100, opacity: 1 }}
  transition={{
    x: { type: 'spring', stiffness: 300 },
    opacity: { duration: 0.3, ease: 'easeIn' }
  }}
/>

Useful Hooks

HookDescription
useAnimation()Imperative animation control — trigger animations programmatically with controls.start(), controls.stop()
useMotionValue()Create a motion value to drive animations without re-renders. Use with useTransform.
useTransform()Map one motion value to another with a range. Essential for scroll-linked animations.
useSpring()Smooth a motion value using spring physics
useScroll()Track scroll position (scrollY, scrollX, scrollYProgress 0-1)
useScrollVelocity()Get scroll speed — useful for skew effects
useInView()Returns true when element is in viewport. Pass once:true for enter-only.
useDragControls()Start drag from a different element (e.g. drag handle)
useReducedMotion()Returns true if user prefers reduced motion — respect accessibility
useAnimate()New in v10 — scope + imperative animate() function for sequences
jsx — useAnimation (imperative control)
import { useAnimation, motion } from 'framer-motion';

function PulseButton() {
  const controls = useAnimation();

  async function handleClick() {
    await controls.start({ scale: 1.2, duration: 0.1 });
    await controls.start({ scale: 1, duration: 0.2 });
    // Can chain sequences with await
  }

  return (
    <motion.button animate={controls} onClick={handleClick}>
      Click
    </motion.button>
  );
}

Common Patterns

jsx — scroll-triggered fade in (useInView)
import { motion, useInView } from 'framer-motion';
import { useRef } from 'react';

function FadeIn({ children, delay = 0 }) {
  const ref = useRef(null);
  const isInView = useInView(ref, { once: true, margin: '-100px' });

  return (
    <motion.div
      ref={ref}
      initial={{ opacity: 0, y: 30 }}
      animate={isInView ? { opacity: 1, y: 0 } : { opacity: 0, y: 30 }}
      transition={{ duration: 0.6, delay, ease: [0.22, 1, 0.36, 1] }}
    >
      {children}
    </motion.div>
  );
}
jsx — marquee / ticker with Framer Motion
import { motion } from 'framer-motion';

function Marquee({ children, speed = 30 }) {
  return (
    <div style={{ overflow: 'hidden', display: 'flex' }}>
      <motion.div
        style={{ display: 'flex', gap: '2rem' }}
        animate={{ x: ['0%', '-50%'] }}
        transition={{ duration: speed, repeat: Infinity, ease: 'linear' }}
      >
        {/* Duplicate content for seamless loop */}
        {children}
        {children}
      </motion.div>
    </div>
  );
}
jsx — reduced motion accessibility
import { useReducedMotion, motion } from 'framer-motion';

function AccessibleAnimation({ children }) {
  const prefersReducedMotion = useReducedMotion();

  return (
    <motion.div
      initial={{ opacity: 0, y: prefersReducedMotion ? 0 : 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: prefersReducedMotion ? 0.01 : 0.6 }}
    >
      {children}
    </motion.div>
  );
}
