JavaScript — Trigger on Event

// Add animation class, remove when done
function animateCSS(element, animation, prefix = 'animate__') {
  return new Promise((resolve) => {
    const el = typeof element === 'string' ? document.querySelector(element) : element;
    const classes = [`${prefix}animated`, `${prefix}${animation}`];
    el.classList.add(...classes);

    function handleEnd() {
      el.classList.remove(...classes);
      el.removeEventListener('animationend', handleEnd);
      resolve('Animation ended');
    }
    el.addEventListener('animationend', handleEnd);
  });
}

// Usage
animateCSS('.my-element', 'bounceOutLeft').then(() => {
  // element has finished animating out
  document.querySelector('.my-element').remove();
});

// Intersection Observer — animate on scroll
const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      entry.target.classList.add('animate__animated', 'animate__fadeInUp');
      observer.unobserve(entry.target);
    }
  });
}, { threshold: 0.1 });

document.querySelectorAll('[data-animate]').forEach(el => observer.observe(el));

AnimXYZ

Composable CSS animation toolkit — combine xyz utilities like Tailwind to create complex animations without writing keyframes. Works with Vue, React, HTML. MIT licensed. Made by Ingram Micro Cloud.

FREE MIT CSS Variables React Vue Composable

animxyz.com →

Install

npm install @animxyz/core
npm install @animxyz/react # React
npm install @animxyz/vue3 # Vue 3
npm install @animxyz/vue # Vue 2

CSS Import

/* In your main CSS or JS entry */
@import "@animxyz/core";

HTML Usage — Compose animations with xyz attribute

<!-- Basic: fade + slide up -->
<div class="xyz-in" xyz="fade up">Animates in</div>

<!-- Combined effects -->
<div class="xyz-in" xyz="fade up big">Big fade up</div>
<div class="xyz-in" xyz="fade left small">Small slide from right</div>
<div class="xyz-in" xyz="fade rotate-right">Fade + rotate</div>
<div class="xyz-in" xyz="fade up-100%">Slide full height</div>

<!-- Direction control -->
<div xyz="in-left out-right">Slides in from left, exits right</div>

<!-- Custom CSS variable overrides -->
<div class="xyz-in" xyz="fade" style="--xyz-translate-x: -42%">Custom offset</div>
<div class="xyz-in" xyz="fade up" style="--xyz-duration: 0.3s; --xyz-delay: 0.5s">Custom timing</div>

<!-- Nested/staggered children -->
<ul class="xyz-in" xyz="fade up stagger">
  <li>Stagger item 1</li>
  <li>Stagger item 2</li>
  <li>Stagger item 3</li>
</ul>

React Usage

import { XyzTransition, XyzTransitionGroup } from "@animxyz/react";
import "@animxyz/core";

// Single element transition
function FadeInCard({ visible }: { visible: boolean }) {
  return (
    <XyzTransition appear xyz="fade up">
      {visible && (
        <div className="card">I animate in and out</div>
      )}
    </XyzTransition>
  );
}

// List with stagger
function AnimatedList({ items }: { items: string[] }) {
  return (
    <XyzTransitionGroup appear xyz="fade up stagger">
      {items.map((item) => (
        <div key={item} className="item">{item}</div>
      ))}
    </XyzTransitionGroup>
  );
}

AutoAnimate

Zero-config, drop-in animation utility. Add one line of code to animate DOM additions, removals, and moves automatically. MIT licensed by FormKit. Works with React, Vue, Solid, Svelte, Angular, and vanilla JS.

FREE MIT React Vue Solid Svelte Vanilla JS

auto-animate.formkit.com →

Install

npm install @formkit/auto-animate

React Hook

import { useAutoAnimate } from '@formkit/auto-animate/react';

// Animated todo list
export function TodoList() {
  const [parent] = useAutoAnimate();
  const [items, setItems] = useState(['Buy groceries', 'Walk the dog']);

  const addItem = () => setItems(prev => [...prev, `Item ${prev.length + 1}`]);
  const removeItem = (i: number) => setItems(prev => prev.filter((_, idx) => idx !== i));

  return (
    <div>
      <button onClick={addItem}>Add item</button>
      <ul ref={parent}>
        {items.map((item, i) => (
          <li key={item} onClick={() => removeItem(i)}>{item}</li>
        ))}
      </ul>
    </div>
  );
}

// Animated accordion/show-hide
export function Accordion({ title, content }: { title: string; content: string }) {
  const [parent] = useAutoAnimate();
  const [open, setOpen] = useState(false);
  return (
    <div ref={parent}>
      <button onClick={() => setOpen(!open)}>{title}</button>
      {open && <p>{content}</p>}
    </div>
  );
}

Motion (Motion One)

Production-grade animation library for the web. Hybrid engine — JavaScript-powered with hardware acceleration via the Web Animations API. Significantly smaller than GSAP. MIT licensed. Available for React, Vue, and vanilla JS. 370+ copy-paste examples.

FREE MIT React Vue Vanilla JS Web Animations API

motion.dev →   Examples →

Install

npm install motion

Vanilla JS — animate()

import { animate, scroll, spring } from "motion";

// Basic animation
animate("#box", { x: 100, opacity: 1 }, { duration: 0.5 });

// Spring physics
animate("#card", { scale: 1 }, {
  type: spring,
  stiffness: 400,
  damping: 20
});

// Keyframes
animate("#circle", {
  x: [0, 100, 0],
  rotate: [0, 180, 360]
}, { duration: 2, repeat: Infinity });

// Scroll-linked animation
scroll(animate("#progress", { scaleX: [0, 1] }));

// Stagger
import { stagger } from "motion";
animate(".item", { opacity: 1, y: [20, 0] }, {
  delay: stagger(0.1),
  duration: 0.4
});

React — motion components

import { motion, AnimatePresence, useMotionValue, useTransform } from "motion/react";

// Basic animated element
function AnimatedCard() {
  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      exit={{ opacity: 0, y: -20 }}
      transition={{ type: "spring", stiffness: 300, damping: 30 }}
      className="card"
      whileHover={{ scale: 1.03 }}
      whileTap={{ scale: 0.97 }}
    >
      Animated card
    </motion.div>
  );
}

// Exit animations
function Modal({ show }: { show: boolean }) {
  return (
    <AnimatePresence>
      {show && (
        <motion.div
          initial={{ opacity: 0, scale: 0.95 }}
          animate={{ opacity: 1, scale: 1 }}
          exit={{ opacity: 0, scale: 0.95 }}
          className="modal"
        >
          Modal content
        </motion.div>
      )}
    </AnimatePresence>
  );
}

// Scroll-linked parallax
import { useScroll, useTransform } from "motion/react";

function ParallaxHero() {
  const { scrollY } = useScroll();
  const y = useTransform(scrollY, [0, 500], [0, -150]);
  return <motion.div style={{ y }} className="hero-bg" />;
}

Tippy.js

Highly customizable tooltip and popover library powered by Popper.js (now Floating UI). Smart positioning, accessible, animated, and themeable. MIT licensed. Works with vanilla JS and has React bindings.

FREE MIT Vanilla JS React Popper.js

tippyjs docs →

Install

npm install tippy.js

Vanilla JS Usage

import tippy from 'tippy.js';
import 'tippy.js/dist/tippy.css';
import 'tippy.js/themes/light.css';

// Basic tooltip
tippy('#myButton', { content: 'Hello!' });

// Full options
tippy('.has-tooltip', {
  content: 'Tooltip content',
  placement: 'top',          // top, bottom, left, right + -start/-end variants
  animation: 'scale',        // fade, scale, shift-away, perspective
  theme: 'light',            // light, light-border, material, translucent
  trigger: 'mouseenter',     // mouseenter, focus, click, manual
  delay: [300, 0],           // [show delay, hide delay]
  arrow: true,
  interactive: true,         // allows hovering the tooltip itself
  appendTo: document.body,
});

// HTML content tooltip
tippy('#btn', {
  content: '<strong>Rich</strong> content',
  allowHTML: true,
  theme: 'light-border',
});

// Programmatic control
const instance = tippy('#el', { content: 'Tip' });
instance[0].show();
instance[0].hide();
instance[0].destroy();
